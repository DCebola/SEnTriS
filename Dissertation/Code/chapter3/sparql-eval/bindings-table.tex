State:
    $bindings$ := {}
    $patterns$ := {}

function setVariables($vars$):
    foreach $var$ in $vars$:
        $bindings$[$var$] := {}
        $patterns$[$var$] := {}

function add($patternID$, $var$, $binding$):
    addBinding($binding$, $var$, $patternID$)
    addPattern($binding$, $var$, $patternID$)

function addPattern($binding$, $var$, $patternID$):
    $varPatternIDs$ := $patterns$[$var$]
    if $varPatternIDs$ == :bot: 
        $varPatternIDs$ := {}
        $varPatternIDs$[$patternID$] := $binding$
        $patterns$[$var$] := $varPatternIDs$
    else:
        $varPatternIDs$[$patternID$] := $binding$

function addBinding($binding$, $var$, $patternID$):
    $varBindings$ = $binding$[$var$]
    if $varBindings$ == :bot:
        $varBindings$ := {}
        savePatternID($varBindings$, $binding$, $patternID$)
        $binding$[$var$] := $varBindings$
    else 
        $varBindingPatternIDs$ = $varBindings$[$binding$]
        if $varBindingPatternIDs$ == :bot:
            savePatternID($varBindings$, $binding$, $patternID$)
        else 
            $varBindingPatternIDs$ := $varBindingPatternIDs$ :U { $patternID$ } 

function savePatternID($varBindings$, $binding$, $patternID$):
    $varBindingPatternIDs$ := {}
    $varBindingPatternIDs$ := $varBindingPatternIDs$ :U { $patternID$ }
    $varBindings$[$binding$] := $varBindingPatternIDs$

function getVariables():
    return $bindings$.getKeys()

function getBindings($var$):
    return $bindings$[$var$]

function getBindings():
    return $bindings$

function getPatternIDs($var$):
    return $patterns$[$var$]

function getPatternIDs():
    return $patterns$

function getPatterns():
    $patterns$ := createLinkedList()
    $vars$ := getVariables()
    $patternIDs$ := {}
    foreach $var$ in $vars$:
        $patternIDs$ := $patternIDs$ :U $patterns$[$var$].getKeys()
    foreach $patternID$ in $patternIDs$:
        $pattern$ := [$\#vars$]
        $i$ := 0
        foreach $var$ in $vars$:
            $binding$ := $patterns$[$var$][$patternID$]
            $pattern$[$i$] := $binding$
            if $binding$ == :bot:
                $i$++
        if $i$ < $\#vars$:
            $patterns$ := $patterns$ :U { $pattern$ }
    return $patterns$

function project($vars$):
    $bindings$.retain($vars$)
    $patterns$.retain($vars$)

function join($other$):
    $vars_{mutual}$ := copy(getVariables())
    $vars_{mutual}$.retain($other$.getVariables())
    return join($vars_{mutual}$, $this$, $other$, "INNER")

function leftOuterJoin($other$): 
    $vars_{mutual}$ := copy(getVariables())
    $vars_{mutual}$.retain($other$.getVariables())
    return join($vars_{mutual}$, $this$, $other$, "LEFT")

function union($other$):
    $vars_l$ := getVariables()
    $vars_r$ := $other$.getVariables()
    $vars$ := copy($vars_l$)
    $vars$ := $vars$ :U $vars_r$ 
    $table_{union}$ := createBindingsTable()
    $table_{union}$.setVariables($vars$)
    copyAllBindings($vars_l$, $this$, $table_{union}$)
    copyAllBindings($vars_r$, $other$, $table_{union}$)
    return $table_{union}$

function minus($other$):
    $vars_{mutual}$ := copy(getVariables())
    $vars_{mutual}$.retain($other$.getVariables())
    $di\textrm{\emph{ff}}$ := {}
    $table_{minus}$ := createBindingsTable()
    $table_{minus}$.setVariables($vars$)
    foreach $var$ : $vars_{mutual}$:
        $bindings_l$ := getBindings($var$)
        $bindings_r$ := $other$.getBindings($var$)
        foreach $(binding, \; patternIDs)$ in $bindings_l$:
            if $bindings_r$[$binding$] == :bot:
                $di\textrm{\emph{ff}}$ := $di\textrm{\emph{ff}}$ :U $patternIDs$
    $varBindings_{RND}$ := getBindings(sample($vars_{mutual}$))
    foreach $(binding, \; patternIDs)$ in $varBindings_{RND}$:
        foreach $patternID$ in $patternIDs$:
            if $patternID \in di\textrm{\emph{ff}}$:
                copyBindings($patternID$, $patternID$, $vars_{mutual}$, $this$, $table_{minus}$)
    return $table_{minus}$
    
function copyBindings($newPatternID$, $oldPatternID$, $vars$, $source$, $target$) 
    foreach $var$ in $vars$
        $binding$ := $source$.getPatternIDs($var$)[$oldPatternID$]
        if $binding$ :neq :bot:
            $target$.add($newPatternID$, $var$, $binding$)
    
function copyAllBindings($vars$, $source$, $target$) 
    foreach $var$ in $vars$:
        $varBindings$ := $source$.getBindings($var$)
        foreach $(binding, \; patternIDs)$ in $varBindings$:
            foreach $patternID$ in $patternIDs$:
                $target$.add($patternID$, $var$, $binding$)

function join($vars_{mutual}$, $table_l$, $table_r$, $type$) 
    $vars_l$ := copy($table_l$.getVariables())
    $vars_r$ := copy($table_r$.getVariables())
    $vars$ := $vars_l$ :U $vars_r$
    $vars_l$ := $vars_l$ / $vars_r$
    $vars_r$ := $vars_r$ / $vars_l$
    $table_{join}$ := createBindingsTable()
    $table_{join}$.setVariables($vars$)
    $var$ := sample($vars_{mutual}$) 
    $bindings_l$ = $table_l$.getBindings($var$)
    $bindings_r$ = $table_r$.getBindings($var$)
    foreach $(binding, \; patternIDs_l)$ in $bindings_l$: 
        $patternIDs_r$ = $bindings_r$[$binding$]
        if $patternIDs_r$ :neq :bot $\:\:$ or type == "LEFT":
            joinPatterns($vars_{mutual}$, $table_l$, $vars_l$, $patternIDs_l$, $table_r$, $vars_r$, $patternIDs_r$, $table_{join}$, $type$)
    return $table_{join}$

function joinPatterns($vars_{mutual}$, $table_l$, $vars_l$, $patternIDs_l$, $table_r$, $vars_r$, $patternIDs_r$, $table_{join}$, $type$) 
    foreach $patternID_l$ in $patternIDs_l$:
        $match$ := false
        if $patternIDs_r$ :neq :bot:
            foreach $patternID_r$ in $patternIDs_r$:
                if equalPatterns($vars_{mutual}$, $table_l$, $patternID_l$, $table_r$, $patternID_r$): 
                    $match$ := true
                    $patternID$ = generateID()
                    copyBindings($patternID$, $patternID_l$, $vars_{mutual}$, $table_l$, $table_{join}$)
                    copyBindings($patternID$, $patternID_l$, $vars_l$, $table_l$, $table_{join}$)
                    copyBindings($patternID$, $patternID_r$, $vars_r$, $table_r$, $table_{join}$)
        if $match$ == false $\:\:$ and type == "LEFT": 
            copyBindings($patternID_l$, $patternID_l$, $vars_{mutual}$, $table_l$, $table_{join}$)
            copyBindings($patternID_l$, $patternID_l$, $vars_l$, $table_l$, $table_{join}$)

function equalPatterns($vars_{mutual}$, $table_l$, $patternID_l$, $table_r$, $patternID_r$) 
    foreach $var$ in $vars_{mutual}$: 
        $binding_l$ = $table_l$.getPatternIDs(var)[$patternID_l$]
        $binding_r$ = $table_r$.getPatternIDs(var)[$patternID_r$]
        if $binding_l$ == :bot $\:\:$ and $binding_r$ :neq :bot
            return false
        elif $binding_l$ :neq :bot $\:\:$ and $binding_r$ == :bot
            return false
        elif $binding_l$ :neq :bot $\:\:$ and $binding_l$ :neq $binding_r$
            return false
    return true
