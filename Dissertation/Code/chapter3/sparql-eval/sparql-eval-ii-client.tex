State:
    $queryEngine$ := createSPARQLQueryEngine()
    $planner$ := :bot

function executeSPARQLQuery($triplestoreID$,$query$,$in\textrm{\emph{f}}erence$,$expansionDepth$,$transitivityDepth$):
    try:
        ($plan$, $construct$, $upload$, $delete$) := planQuery($query$, $triplestoreID$, $in\textrm{\emph{f}}erence$, $expansionDepth$, $transitivityDepth$) 
        $queryType$ := $planner$.getQueryType()
        if $queryType \in \{$ $SELECT$, $CONSTRUCT$, $ASK$, $DESCRIBE$, $MODIFY$, $DELETE\_WHERE$ $\}$:
            $sparqlResult$ := Triplestore.evaluateQuery($triplestoreID$, $plan$)
        return generateFinalResult($triplestoreID$, $queryType$, $plan$.$vars$, $sparqlResult$.$bindings$, $construct$, $upload$, $delete$)
    catch Exception:
        return :bot

function planQuery($query$, $triplestoreID$, $in\textrm{\emph{f}}erence$, $expansionDepth$, $transitivityDepth$):
    $planner$ := createSPARQLQueryPlanner()
    if $in\textrm{\emph{f}}erence$ == true:
        $schema$ := Triplestore.fecthSchema($triplestoreID$)
        $ontology$ := createOntology()
        $ontology$ := execInference($schema$, $in\textrm{\emph{f}}erence$, $expansionDepth$, $transitivityDepth$)
        $planner$.setOntology($ontology$)   
    return ($queryEngine$.getQueryPlan($planner$, $query$), $planner$.getConstructTemplate(), $planner$.getUploadTemplate(), $planner$.getDeleteTemplate())
