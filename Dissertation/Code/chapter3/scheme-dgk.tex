State:
    ($K_{DGK_{priv}}$, $K_{DGK_{pub}}$) := generateDGKKeyPair()
    $eqTags$ := {}
    $eq_{last}$ := 0

function ReInitialize($master$, $rnd$, $KeyPair_{DGK}$, $iv$, $schema$):
    loadState($master$, $rnd$, $iv$, $schema$)
    ($K_{DGK_{priv}}$, $K_{DGK_{pub}}$) := $KeyPair_{DGK}$

function getSecrets():
    return $(K_{MASTER}, \; K_{RND}, \; K_{DGK_{priv}}, \; K_{DGK_{pub}}, \; iv_{DET}, \; keyword_{schema})$

function encryptContextualData():
    foreach $(keyword, \; frequency)$ in $keywordFrequencies$:
        $trapdoor$ := E($derivedKeys$[$keyword$], $keyword$, $iv_{ZERO}$)
        $encryptedNodes$[$trapdoor$] := E($K_{RND}$, $keywordFrequencies$[$keyword$])
    foreach $(node, \; eqTag)$ in $eqTags$:
        $trapdoor$ := E($derivedKeys$[$node$], $node$, $iv_{ZERO}$)
        $encryptedNodes$[$trapdoor$] := E($K_{RND}$, $eqTag$)
    return $encryptedNodes$
        
function encryptNode($node$, $keyword$):
    $freq$ := incrementKeywordFrequency($keyword$)
    $trapdoor$ := E(deriveKey($keyword$), $keyword$, $freq$)
    $enc_{eq_{tag}}$ := E($K_{DGK_{pub}}$, getEqTag(node))
    $encryptedNodes$[$trapdoor$] := $enc_{eq_{tag}}$
    $encryptedNodes$[$enc_{eq_{tag}}$] := E($K_{RND}$, $node$)
    return $freq$

function getEqTag($node$):
    $eq_{tag}$ := $eqTags$[$node$]
    if $eq_{tag}$ == :bot:
        $eq_{tag}$ := $eq_{last}$ + 1
        $eqTags$[$node$] := $eq_{tag}$
    if ($eq_{tag}$ > $eq_{last}$)
        $eq_{last}$ := $eq_{tag}$
    return $eq_{tag}$;
    
function setEqTags($value$):
    $eqTags$ := $value$

function getDETKey():
    return $K_{DET}$

function getIvDET():
    return $iv_{DET}$

function getLastEqTag():
    return $eq_{last}$

function getEqKey():
    return ($K_{DGK_{priv}}.p$, $K_{DGK_{priv}}.vp$, $K_{DGK_{pub}}.n$, $K_{DGK_{pub}}.u$)

function getPubDGK()
    return ($K_{DGK_{pub}}.n$, $K_{DGK_{pub}}.g$, $K_{DGK_{pub}}.h$, $K_{DGK_{pub}}.u$)

function clearEqTags():
    $eqTags$ := {}
