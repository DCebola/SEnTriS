function search($triplestoreID$, $varPattern$, $s$, $p$, $o$):
    switch($varPattern$):
        case "S": return fetchBindings(generateIndex($triplestoreID$, "S", $p$, $o$), $s$)
        case "P": return fetchBindings(generateIndex($triplestoreID$, "P", $s$, $o$), $p$)
        case "O": return fetchBindings(generateIndex($triplestoreID$, "P", $s$, $p$), $o$)
        case "PO": return fetchBindings(generateIndex($triplestoreID$, "PO", $s$), $p$, $o$)
        case "SO": return fetchBindings(generateIndex($triplestoreID$, "SO", $p$), $s$, $o$)
        case "SP": return fetchBindings(generateIndex($triplestoreID$, "SP", $o$), $s$, $p$)

function fetchBindings($index$, $var$):
    $table$ := createBindingsTable()
    $table$.setVariables({$var$})
    foreach $node$ in $triplestores$[$index$]:
        $table$.add(generateID(), $var$, $node$)
    return $table$

function fetchBindings($index$, $var_1$, $var_2$):
    $table$ := createBindingsTable()
    $table$.setVariables({$var_1$, $var_2$})
    foreach $nodePair$ in $triplestores$[$index$]:
        $patternID$ := generateID()               
        $nodes$ = $nodePair$.split($separator_{III}$)
        $table$.add($patternID$, $var_1$, $nodes$[0])
        $table$.add($patternID$, $var_2$, $nodes$[1])
    return $table$

function parseNode($n$):
    if $n \in I$:            
        return "S" || $separator_{I}$ || $n.value$  // IRI
    elif $n \in L$:
        return "L" || $separator_{I}$ || $n.value$ || $separator_{I}$ || $n.datatype$
    elif $n \in B$:
        return "B" || $separator_{I}$ || $n.value$    // identifier of the blank node
    return $\bot$

function parseKeyword($n$):
    if $n \in I$:            
        return "S" || $separator_{I}$ || $n.value$   
    elif $n \in L$:
        return "L" || $separator_{I}$ || $n.value$ || $separator_{I}$ || $n.datatype$
    elif $n \in B$:
        return "BLANK"
    return $\bot$

function generateNode($parsedValue$):
    $(prefix, \; value, \; datatype)$ := $parsedValue$.split($separator_{I}$);
    if prefix == "BLANK":
        return createBlankNode($value$)
    elif prefix == "S":
        return createIRINode($value$);
    else
        return createLiteralNode($value$, $datatype$);
