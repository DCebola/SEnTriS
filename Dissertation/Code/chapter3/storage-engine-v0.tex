function search($triplestoreID$, $varPattern$, $s$, $p$, $o$):
    switch($varPattern$):
        case "S": return fetchBindings($triplestoreID$, generateIndex("S", $p$, $o$), $s$)
        case "P": return fetchBindings($triplestoreID$, generateIndex("P", $s$, $o$), $p$)
        case "O": return fetchBindings($triplestoreID$, generateIndex("P", $s$, $p$), $o$)
        case "PO": return fetchBindings($triplestoreID$, generateIndex("PO", $s$), $p$, $o$)
        case "SO": return fetchBindings($triplestoreID$, generateIndex("SO", $p$), $s$, $o$)
        case "SP": return fetchBindings($triplestoreID$, generateIndex("SP", $o$), $s$, $p$)
        default: throw NotImplementedException()

function fetchBindings($triplestoreID$, $index$, $var$):
    $table$ := createBindingsTable()
    $table$.setVariables({$var$})
    foreach $node$ in $triplestores$[$triplestoreID$][$index$]:
        $table$.add(generateID(), $var$, $node$)
    return $table$

function fetchBindings($triplestoreID$, $index$, $var_1$, $var_2$):
    $table$ := createBindingsTable()
    $table$.setVariables({$var_1$, $var_2$})
    foreach $nodePair$ in $triplestores$[$triplestoreID$][$index$]:
        $patternID$ := generateID()               
        $nodes$ = $nodePair$.split($SEPARATOR_{III}$)
        $table$.add($patternID$, $var_1$, $nodes$[0])
        $table$.add($patternID$, $var_2$, $nodes$[1])
    return $table$

function parseNode($n$):
    if $n \in I$:            
        return "S" || $SEPARATOR_{I}$ || $n.value$  // IRI
    elif $n \in L$:
        return "L" || $SEPARATOR_{I}$ || $n.value$ || $SEPARATOR_{I}$ || $n.datatype$
    elif $n \in B$:
        return "B" || $SEPARATOR_{I}$ || $n.value$    // identifier of the blank node
    return $\bot$

function parseKeyword($n$):
    if $n \in I$:            
        return "S" || $SEPARATOR_{I}$ || $n.value$   
    elif $n \in L$:
        return "L" || $SEPARATOR_{I}$ || $n.value$ || $SEPARATOR_{I}$ || $n.datatype$
    elif $n \in B$:
        return "BLANK"
    return $\bot$

function generateNode($parsedValue$):
    $(prefix, \; value, \; datatype)$ := $parsedValue$.split($SEPARATOR_{I}$);
    if $pre\textrm{\emph{f}}ix$ == "BLANK":
        return createBlankNode($value$)
    elif $pre\textrm{\emph{f}}ix$ == "S":
        return createIRINode($value$)
    else
        return createLiteralNode($value$, $datatype$)
