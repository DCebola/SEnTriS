function executeSPARQLQuery($triplestoreID$, $query$):
    try:
        ($planner$, $plan$) := planQuery($query$) 
        $queryType$ := $planner$.getQueryType()
        if $queryType \in \{$ $SELECT$, $CONSTRUCT$, $ASK$, $DESCRIBE$, $MODIFY$, $DELETE\_WHERE$ $\}$:
            $sparqlResult$ := Triplestore.evaluateQuery($triplestoreID$, $plan$)
        return generateFinalResult($triplestoreID$, $queryType$, $sparqlResult$, $construct$, $upload$, $delete$)
    catch Exception:
        return :bot

# SPARQL-EVAL.planQuery(query)
function planQuery(query):
    $planner$ := createSPARQLQueryPlanner()
    return ($planner$, $queryEngine$.getQueryPlan($planner$, $query$))

# SPARQL-EVAL.generateFinalResult($triplestoreID$, $queryType$, $sparqlResult$, $construct$, $upload$, $delete$):
function generateFinalResult($triplestoreID$, $queryType$, $variables$, $sparqlResult$, $construct_{template}$, $upload_{template}$, $delete_{template}$)
    if $queryType$ == "SELECT":
        return generateSELECTResults($variables$, $sparqlResult$)
    elif $queryType$ == "CONSTRUCT":
        return generateCONSTRUCTResults($construct_{template}$, $sparqlResult.bindings$)
    elif $queryType$ == "ASK":
        return generateASKResults($sparqlResult$.$bindings$)
    elif $queryType$ == "DESCRIBE":
        return generateDESCRIBEResults($variables$, $sparqlResult$)
    else:
        return applyModifications($triplestoreID$, $sparqlResult$.$bindings$, $upload_{template}$, $delete_{template}$)   

function applyModifications($triplestoreID$, $bindings$, $upload_{template}$, $delete_{template}$):
    $triples_{upload}$ := {}
    $triples_{delete}$ := {}
    if $queryType$ == "MODIFY":
        $triples_{upload}$ := generateTriplesFromBindings($upload_{template}$, $bindings$) #foreach binding, map values to variables and output the set of generated triples
    $triples_{delete}$ := generateTriplesFromBindings($delete_{template}$, $bindings$) #foreach binding, map values to variables and output the set of generated triples
    else if (queryType == INSERT_DATA)
        $triples_{upload}$ := $upload_{template}$
    else if (queryType == DELETE_DATA)
        $triples_{delete}$ := $delete_{template}$
    try:
        Triplestore.deleteTriples($triplestoreID$, $triples_{delete}$)
        Triplestore.saveTriples($triplestoreID$, $triple_{upload}$)
    catch Exception:
        return :bot
    return "SUCCESS"

function generateASKResults($bindings$):
    return $\#bindings \; > \; 0$

function generateCONSTRUCTResults($construct_{template}$, $bindings$):
    return generateGraphFromBindings($construct_{template}$, $bindings$) #foreach binding, map values to variables and output a graph with all the generated triples

function generateSELECTResults($variables$, $bindings$):
    return generateJSONQueryResult($variables$, $bindings$) # map bindings collection to a well-formed JSON, as per the official specification

private Response generateDESCRIBEResults($variables$, $bindings$) :
    frequencies := {}
    foreach $var$ in $variables$:
        $frequencies$[$var$] := 0
    foreach $binding$ in $bindings$:
        foreach $(var, \; node)$ in $binding$.getKeys(): 
            $frequencies$[$var$] := $frequencies$[$var$] + 1
    return generateJSONFrequencyTable($frequencies$) # format the collection of variable frequencies as JSON
