function setContext($k$, $storage$):
    $key$ := $k$
    $storageEngine$ := $storage$
    $execID$ := generateID()

function exec($job$):
    if $job$ is SEC_SEARCH:
        $searches$ := $job$.getSearches()
        foreach $(var, \; searchID)$ in $searches$
            $searchIDs$ :=  $searchIDs$ :U { $searchID$ }
        return $storageEngine$.search($key$, $job$.getVariables(), $searches$, $execID$)
    elif $job$ is EMPTY:
        return createBindingsTable($job$.vars)
    throw JobInstanceException()

function generateBindings($collector$, $jobResults$):
    vars = $jobResults$.getVariables()
    values := {}
    $eqTags$ := $storageEngine$.getEqTags($execID$)
    foreach $patterns$ in $jobResults$.getPatterns():
        $binding$ := {}
        $i$ := 0
        foreach $encryptedBinding$ in $patterns$:
            if $encryptedBinding$ :neq :bot:
                $binding$[$vars$[$i$]] := $eqTags$[$encryptedBinding$]
            $i$++
        $collector$.add($binding$)
    $storageEngine$.getEqTags($execID$)
    return $collector$
