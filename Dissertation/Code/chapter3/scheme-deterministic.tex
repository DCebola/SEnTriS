    loadState($master$, $rnd$, $schema$)
    $K_{DET}$ := $det$
    $iv_{DET}$ := $iv$

function encryptKeywordInfo():
    foreach $keyword$ in $keywordFrequencies$:
        $trapdoor$ := ENC($derivedKeys$[$keyword$], $keyword$, $iv_{ZERO}$)
        $encryptedNode$[$trapdoor$] := ENC($K_{RND}$, $keywordFrequencies$[$keyword$])

function encryptNode($node$, $keyword$):
    $freq$ := incrementKeywordFrequency($keyword$)
    $trapdoor$ := ENC(deriveKey($keyword$), $keyword$, $freq$)
    $encryptedNode$[$trapdoor$] := ENC($K_{RND}$, ENC($K_{DET}$, $node$, $iv_{DET}$))
    return $freq$
