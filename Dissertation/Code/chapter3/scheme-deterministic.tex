function ReInitialize($master$, $rnd$, $det$, $iv$, $schema$):
    loadState($master$, $rnd$, $iv$, $schema$)
    $K_{DET}$ := $det$

function getSecrets():
    return $(K_{MASTER}, \; K_{RND}, \; K_{DET}, \; iv_{DET}, \; keyword_{schema})$

function encryptContextualData():
    foreach $(keyword, \; frequency)$ in $keywordFrequencies$:
        $trapdoor$ := E($derivedKeys$[$keyword$], $keyword$, $iv_{ZERO}$)
        $encryptedNodes$[$trapdoor$] := E($K_{RND}$, $\textrm{\emph{f}}requency$)
    return $encryptedNodes$

function encryptNode($node$, $keyword$):
    $\textrm{\emph{f}}req$ := incrementKeywordFrequency($keyword$)
    $trapdoor$ := E(deriveKey($keyword$), $keyword$, $\textrm{\emph{f}}req$)
    $encryptedNodes$[$trapdoor$] := E($K_{RND}$, E($K_{DET}$, $node$, $iv_{DET}$))
    return $\textrm{\emph{f}}req$
 
function decryptNode($node$):
    return D($K_{RND}$, D($K_{DET}$, $node$, $iv_{DET}$))

function decryptBinding($ct$):
    return D($K_{DET}$, $ct$)

function getDETKey():
