    loadState($master$, $rnd$, $iv$, $schema$)

function encryptKeywordInfo():
    foreach $keyword$ in $keywordFrequencies$:
        $trapdoor$ := E($derivedKeys$[$keyword$], $keyword$, $iv_{ZERO}$)
        $encryptedNode$[$trapdoor$] := E($K_{RND}$, $keywordFrequencies$[$keyword$])

function encryptNode($node$, $keyword$):
    $freq$ := incrementKeywordFrequency($keyword$)
    $trapdoor$ := E(deriveKey($keyword$), $keyword$, $freq$)
    $encryptedNode$[$trapdoor$] := E($K_{RND}$, E($K_{DET}$, $node$, $iv_{DET}$))
    return $freq$
