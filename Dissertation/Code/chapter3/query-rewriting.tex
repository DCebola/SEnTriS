function extractVarsOfBinaryJob($jobs$, $job = (jobID, \; jobID_l, \; jobID_r)$): 
    $vars$ := {}
    $jobs_{queue}$ := createQueue()
    $jobs_{queue}$.push($jobs$[$jobID_l$])
    $jobs_{queue}$.push($jobs$[$jobID_r$])
    while $\#jobs_{queue} > 0$: 
        $job_{next}$ := $jobs_{queue}$.poll()
        if $job_{next}$ is SEARCH:
            $vars$ := $vars$ :U extractVars(searchJob)
        elif $job_{next} = (jobID_{next}, \; l, \; r)$ is BINARY:
            $jobs_{queue}$.push($jobs$[$l$])
            $jobs_{queue}$.push($jobs$[$r$])
    return $vars$

function pushUnion($left$, $right$, $jobs$, $jobIDs$): 
    $unionPattern$ := "union" || $left$ || $right$
    $jobID$ := $jobIDs$[$unionPattern$]
    if $jobID$ == :bot: 
        $job_{MINUS}$ := createMinusJob(generateID(), $right$, $left$)
        $jobID$ := generateID()
        $job_{UNION}$ := createUnionJob($jobID$, $left$, $job_{MINUS}$.jobID)
        $jobs$[$jobID$] := $job_{UNION}$
        $jobIDs$[$jobID$] := $unionPattern$
        pushJob($job_{MINUS}$)
        pushJob($job_{UNION}$)
    return $jobID$

private String pushJoin($left$, $right$, $jobs$, $jobIDs$):
    $joinPattern$ := "join" || $left$ || $right$
    $jobID$ := $jobIDs$[$joinPattern$]
    if $jobID$ == :bot:
        $jobID$ = generateID()
        $job_{JOIN}$ = new JoinJob(jobID, left, right)
        $jobs$[$jobID$] := $job_{JOIN}$
        $jobIDs$[$joinPattern$] := $jobID$
        pushJob($job_{JOIN}$)
    return $jobID$

function generateSearchJobs($op_{OpBGP}$):
    $patterns$ = $op_{OpBGP}$.getPattern()
    $jobIDs$ = {}
    $jobs$ = {}
    $jobVars$ = {}
    foreach $t = (s, p, o)$ in $patterns$:
        $jobID$ = pushSearch($s$, $p$, $o$, $jobs$, $jobIDs$)
        if $ontology$ :neq :bot:
            if ($p$ == "rdf:type") and ($o$ is not VARIABLE):
                $jobID$ = expandClass($jobID$, $s$, $o$, 0, $jobs$, $jobIDs$)
            elif $p$ is not VARIABLE:
                $jobID$ = expandProperty($jobID$, $s$, $p$, $o$, 0, $jobs$, $jobIDs$, true, true)
        $job$ := $jobs$[$jobID$]
        if $job$ is SEARCH:
            $vars_{job}$ := extractVars($job$)
            $variables$ := $variables$ :U $vars_{job}$
            $jobVars$[$jobID$] := $vars_{job}$
        elif $job$ is BINARY:
            $jobVars$[$jobID$] := extractVarsOfBinaryJob($jobs$, $job$)
    if $\#patterns$ > 1:
        generateRandomJoinPipeline($op_{OpBGP}$, $jobVars$)
    else:
        $operations_{parsed}$.put($op_{OpBGP}$, $jobID$)

function expandClass($jobID$, $s$, $o$, $depth$, $jobs$, $jobsIDs$):
    if $depth$ == $ontology$.getMaximumExpansionDepth():
        return $jobID$
    $jobID$ := expandClassDisjunction($ontology$.getSubClasses($o$), $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
    $jobID$ := expandClassDisjunction($ontology$.getEquivalentClasses($o$), $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
    $jobID$ := expandRestriction($ontology$.getRestriction($o$), $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
    $jobID$ := expandClassDisjunction($ontology$.getIntersectionWhereClassIsOperand($o$), $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
    $intersection$ := $ontology$.getIntersection($o$)
    if (!intersection.isEmpty())
        $jobID$ := expandClassConjunction($intersection$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
    return $jobID$

function expandRestriction($restriction = (type, \; property, \; value)$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$):   
    if $restriction$ :neq :bot:
        if $type$ == "owl:hasValue":
            $left$ := pushSearch(createVariable(parse($property$)), "rdf:type", $value$, $jobs$, $jobsIDs$)
            $right$ := expandProperty(pushSearch($s$, $property$, $value$, jobs, jobsIDs), $s$, $property$, $value$, $depth$ + 1, $jobs$, $jobsIDs$, true, true)
            $jobID$ := pushUnion($left$, $right$, $jobs$, $jobsIDs$)
        elif $type \in \{$"owl:isSomeValuesFrom", "owl:allValuesFrom"$\}$:
            $var$ := createVariable(parse($value$))
            $right$ := pushSearch($s$, $property$, $var$, $jobs$, $jobsIDs$)
            $left$ := pushSearch($var$, "rdf:type", $value$, $jobs$, $jobsIDs$)
            $right_{expanded}$ := expandProperty($right$, $s$, $property$, $var$, $depth$ + 1, $jobs$, $jobsIDs$, true, true)
            $left_{expanded}$ := expandClass($left$, $var$, $value$, $depth$ + 1, $jobs$, $jobsIDs$)
            $join$ := pushJoin($right_{expanded}$, $left_{expanded}$, $jobs$, $jobsIDs$)
            $jobID$ := pushUnion($jobID$, $join$, $jobs$, $jobsIDs$)
    return $jobID$

function expandClassConjunction($classes$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$):
    foreach $c$ in $classes$:
        $search$ := pushSearch($s$, "rdf:type", $c$, $jobs$, $jobsIDs$)
        $left$ = expandClass($search$, $s$, $c$, $depth$ + 1, $jobs$, $jobsIDs$)
        if $right$ :neq :bot:
            $right$ := pushJoin($right$, $left$, $jobs$, $jobsIDs$)
        else:
            $right$ := $left$
    if $right$ == :neq:
        return pushUnion($jobID$, $right$, $jobs$, $jobsIDs$)
    return $jobID$

function expandClassDisjunction($classes$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$):
    foreach $c$ in $classes$:
        $search$ := pushSearch($s$, "rdf:type", $c$, $jobs$, $jobsIDs$)
        $left$ = expandClass($search$, $s$, $c$, $depth$ + 1, $jobs$, $jobsIDs$)
        if $right$ :neq :bot:
            $right$ := pushUnion($right$, $left$, $jobs$, $jobsIDs$)
        else:
            $right$ := $left$
    if $right$ == :neq:
        return pushUnion($jobID$, $right$, $jobs$, $jobsIDs$)
    return $jobID$

function expandProperty($jobID$, $s$, $p$, $o$, $depth$, $jobs$, $jobsIDs$, $symmetric$, $inverseOf$):
    if $depth$ == $ontology$.getMaximumExpansionDepth():
        return $jobID$
    if $symmetric$ == true and $ontology$.isSymmetric($p$):
        $jobID$ := pushUnion($jobID$, pushSearch($o$, $p$, $s$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
        $jobID$ := expandProperty($jobID$, $o$, $p$, $s$, $depth$ + 1, $jobs$, $jobsIDs$, false, $inverseOf$)
    if $ontology$.isTransitive($p$):
        $var$ := createVariable(parse(p) || 0)
        $left$ := pushSearch($s$, $p$, $var$, $jobs$, $jobsIDs$)
        $join$ := pushJoin($left$, pushSearch($var$, $p$, $o$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
        $jobID$ := pushUnion($jobID$, $join$, $jobs$, $jobsIDs$)
        for $i$ in range(0, $ontology$.getTransitivityDepth())
            for $j$ in range(0, $i$ + 1):
                if $j$ == $i$: 
                    $var_{next}$ = createVariable(parse(p) || $j$)
                    $left$ := pushJoin($left$, pushSearch($var$, $p$, $var_{next}$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
                    $var$ = $var_{next}$
            $join$ := pushJoin($left$, pushSearch($var$, $p$, $o$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
            $jobID$ := pushUnion($jobID$, $join$, $jobs$, $jobsIDs$)
    
    $jobID$ := expandProperties($ontology$.getSubProperties($p$), $jobID$, $s$, $o$, $depth$, $jobs$, $jobsIDs$, $symmetric$, $inverseOf$)
    $jobID$ := expandProperties($ontology$.getEquivalentProperties($p$), $jobID$, $s$, $o$, $depth$, $jobs$, $jobsIDs$, $symmetric$, $inverseOf$)
    if $inverseOf$ == true:
        $jobID$ := expandProperties($ontology$.getInverseOf($p$), $jobID$, $o$, $s$, $depth$, $jobs$, $jobsIDs$, $symmetric$, false)
    return $jobID$
