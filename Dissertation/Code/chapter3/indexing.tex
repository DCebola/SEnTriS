State:
    T := (triplestoreID, nodes := {}, acl)

function parseNode($n$):
    if $n \in I$:            
        return "S" || $n.value$   
    elif $n \in L$:
        return "L" || $n.value$ || $n.datatype$
    elif $n \in B$:
        return "B" || $n.id$
    return $\bot$

function parseKeyword($n$):
    if $n \in I$:            
        return "S" || $n.value$   
    elif $n \in L$:
        return "L" || $n.value$ || $n.datatype$
    elif $n \in B$:
        return "BLANK"
    return $\bot$

function indexTriple($t = (s, p, o)$):
    $parsed_s$ = parseNode($s$)
    $parsed_p$ = parseNode($p$)
    $parsed_o$ = parseNode($o$)
    $keyword_s$ = parseKeyword($s$)
    $keyword_p$ = parseKeyword($p$)
    $keyword_o$ = parseKeyword($o$)
    $T.nodes$["PO" || $keyword_s$] := $T.nodes$["PO" || $keyword_s$] :U {$parsed_p$ || $parsed_o$}
    $T.nodes$["SO" || $keyword_p$] := $T.nodes$["SO" || $keyword_p$] :U {$parsed_s$ || $parsed_o$}
    $T.nodes$["SP" || $keyword_o$] := $T.nodes$["SP" || $keyword_o$] :U {$parsed_s$ || $parsed_p$}
    $T.nodes$["S" || $keyword_p$ || $keyword_o$] := $T.nodes$["S" || $keyword_p$ || $keyword_o$] :U {$parsed_s$}
    $T.nodes$["P" || $keyword_s$ || $keyword_o$] := $T.nodes$["P" || $keyword_s$ || $keyword_o$] :U {$parsed_p$}
    $T.nodes$["O" || $keyword_s$ || $keyword_p$] := $T.nodes$["O" || $keyword_s$ || $keyword_p$] :U {$parsed_o$}
