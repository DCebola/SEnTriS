// $\{\textrm{SUCCESS}, \bot\} \leftarrow \textrm{SSE-TP}.CreateTriplestore(triplestoreID, \; [t_0, t_1, ..., t_n])$
function createTriplestore($triplestoreID$, $triples_{ABox}$):
    $scheme$ := createEncryptionScheme() 
    $secrets$[$triplestoreID$] := $scheme$.getSecrets()
    return executeUpdate($triplestoreID$, {}, $triples_{ABox}$, $scheme$)

// $\{\textrm{SUCCESS}, \bot\} \leftarrow \textrm{SSE-TP}.DeleteTriplestore(triplestoreID, \; schema)$
function deleteTriplestore($triplestoreID$, $schema$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    else:
        if $schema$ == true:
            return deleteTBox($triplestoreID$, $schema$)
        return Triplestore.delete($triplestoreID$)

// $\{\textrm{SUCCESS}, \bot\} \leftarrow \textrm{SSE-TP}.Upload(triplestoreID, \; [t_0, t_1, ..., t_n], \; schema)$
function upload($triplestoreID$, $triples$, $schema$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    $scheme$ := createEncryptionScheme()
    $scheme$.ReInitialize($secrets$[$triplestoreID$])
    if $schema$ == true:
        return uploadTBox($triplestoreID$, $triples$, $scheme$)
    return executeUpdate($triplestoreID$, {}, $triples$, $scheme$)

// $\{\textrm{SUCCESS}, \bot\} \leftarrow \textrm{SSE-TP}.Delete(triplestoreID, \; [t_0, t_1, ..., t_n])$
function delete($triplestoreID$, $triples$):

// $\{\textrm{SUCCESS}, \bot\} \leftarrow \textrm{SSE-TP}.UpdateTriplestore(triplestoreID, \; triples_{delete}, \; triples_{upload})$
function updateTriplestore($triplestoreID$, $triples_{delete}$, $triples_{upload}$):

    $scheme$ := createEncryptionScheme()
    $scheme$.ReInitialize($secrets$[$triplestoreID$])

function executeUpdate($triplestoreID$, $triples_{delete}$, $triples_{upload}$):
    $deleteIDs$ := batchExecute($triplestoreID$, $scheme$, $triples_{delete}$, "DELETION")
    $uploadIDs$ := batchExecute($triplestoreID$, $scheme$, $triples_{upload}$, "UPLOAD")
    return Triplestore.update($triplestoreID$, $uploadIDs$, $deleteIDs$)

function batchExecute($triplestoreID$, $scheme$, $triples$, $opType$):
    $batchIDs$ := createLinkedList()
    while $\#triples > 0$: 
        $batch$ := createLinkedList()
        for $i$ in range(0, $BATCH\_SIZE$):
            if $\#triples > 0$:
                $batch$.add($triples$.removeFirst())
        if $\#batch > 0$:
            $batchID$ := batch($triplestoreID$, $scheme$, $batch$, $opType$)
            if $batchID$ == :bot:
                throw BatchException()
            $batchIDs$.add($batchID$)
        $scheme$.clearNodes()
    $scheme$.clearFrequencies()
    return $batchIDs$

function batch($triplestoreID$, $scheme$, $batch$, $opType$):
    if $opType$ == "UPLOAD":
        return prepareUpload($triplestoreID$, $batch$, $scheme$)
    elif $opType$ == "DELETION":
        return prepareDelete($triplestoreID$, $batch$, $scheme$)
    else:
        throw IllegalStateException()
        
function prepareUpload($triplestoreID$, $triples$, $scheme$):
    $keywords$ := [$\#triples$]
    $i$ := 0
    foreach $t$ in $triples$:
        $keyword$ := extractKeyword($triplePattern$) 
        if $keyword$ :neq :bot:
            $keywords$[$i$] := $keyword$
            $i$++
    $frequencies$ := fetchKeywordsFrequencies($triplestoreID$, $keywords$, $scheme$)
    $scheme$.setKeywordFrequencies($frequencies$)
    $encryptedNodes$ := $scheme$.encrypt($triples$, false)
    return Triplestore.prepareUpload($triplestoreID$, $encryptedNodes$)

function prepareDelete($triplestoreID$, $triples$, $scheme$):
    $keywordsTrapdoors$ := $scheme$.generateTriplePatternTrapdoors($triples$)
    $encryptedFrequencies$ := Triplestore.search($triplestoreID$, $triplePatternTrapdoors$)
    $keywords$ := createArrayList($\#keywordsTrapdoors$.getKeys())
    $totalTrapdoors$ := 0
    foreach $keywordTrapdoors$ in $keywordsTrapdoors$.getValues():
        $totalTrapdoors$ := $totalTrapdoors$ + $\#keywordTrapdoors$
    $trapdoors$ := createArrayList($totalTrapdoors$)
    foreach $keyword$ in $keywords$:
        foreach $keywordTrapdoors$ in $keywordsTrapdoors$[$keyword$]:
            foreach $trapdoor$ in $keywordTrapdoors$:
                $trapdoors$.add($trapdoor$)
    $encryptedFrequencies$ := Triplestore.search($triplestoreID$, $trapdoors$)
    $offset$ := 0
    $deletions$ := createLinkedList()
    foreach $keyword$ in $keywords$:
        $length$ := $\#keywordsTrapdoors$[$keyword$]
        for $i$ in range($offset$, $offset$ + $length$) 
            $encryptedFrequency$ := $encryptedFrequencies$.get($i$)
            if $encryptedFrequency$ :neq :bot:
                $frequency$ := $scheme$.decryptContextualData($encryptedFrequency$)
                $deletions$.add($scheme$.generateTrapdoor($keyword$, $frequency$))
                $deletions$.add($trapdoors$.get($i$))
        $offset$ := $offset$ + $length$
    return Triplestore.prepareDelete($triplestoreID$, $trapdoors$)

function uploadTBox($triplestoreID$, $triples$, $scheme$):
    $keyword$ := $scheme$.getSchemaKeyword()
    $frequencies$ := fetchKeywordsFrequencies($triplestoreID$, [$keyword$], $scheme$)
    $deletionIDs$ := {}
    if $frequencies[keyword] > 0$:
        $deletionIDs$ := batchExecuteDeleteTBox($triplestoreID$, $frequencies$[$keyword$], $scheme$)
    $uploadIDs$ := batchExecuteUploadTBox($triplestoreID$, $triples$, $frequencies$[$keyword$], $scheme$)
    return Triplestore.update($triplestoreID$, $uploadIDs$, $deleteIDs$)

function deleteTBox($triplestoreID$, $triples$, $scheme$):
    $keyword$ := $scheme$.getSchemaKeyword()
