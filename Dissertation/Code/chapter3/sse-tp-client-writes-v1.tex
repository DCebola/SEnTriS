function deleteAll($triplestoreID$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    else:
        Triplestore.deleteAll($triplestoreID$)

function upload($triplestoreID$, $triples$, $schema$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    $scheme$ := createEncryptionScheme()
    $scheme$.loadState($secrets$[$triplestoreID$])
    if $schema$ == false:
        return uploadABox($triplestoreID$, $triples$, $scheme$)
    else:
        return updateNodes($triplestoreID$, $triples$, $scheme$, true)

function batchExecute($triplestoreID$, $scheme$, $triples$, $opType$):
    $batchIDs$ := createLinkedList()
    while $\#triples > 0$: 
        $batch$ := createLinkedList()
        for $i$ in range(0, $BATCH_SIZE$):
            if $\#triples > 0$:
                $batch$.add($triples$.removeFirst())
        if $\#batch > 0$:
            $batchID$ := batch($triplestoreID$, $scheme$, $batch$, $opType$)
            if $batchID$ == :bot:
                throw BatchException()
            $batchIDs$.add($batchID$)
        $scheme$.clearNodes()
    return $batchIDs$

function batch($triplestoreID$, $scheme$, $batch$, $opType$):
    if $opType$ == "UPLOAD":
        return prepareABoxUpload($triplestoreID$, $batch$, $scheme$)
    elif $opType$ == "DELETION":
        return prepareABoxDelete($triplestoreID$, $batch$, $scheme$)
    else:
        throw IllegalStateException()
