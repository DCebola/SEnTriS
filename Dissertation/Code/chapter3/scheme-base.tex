    $K_{MASTER}$ := generateKey() 
    $K_{RND}$ := generateKey() 
    $iv_{DET}$ := generateRandomIV()
    $keyword_{schema}$ := generateRandomString()
    $iv_{ZERO}$ := generateZeroFilledIV()
    $encryptedNodes$ := {}
    $keywordFrequencies$ := {}
    $derivedKeys$ := {} 
    
// $E(T)$ := $\textrm{SSE-TP}.ReInitialize(K_{MASTER}, \; K_{RND}, \; keyword_{schema})$
function loadState($master$, $rnd$, $iv$, $schema$):  
    $K_{MASTER}$ := $master$ 
    $K_{RND}$ := $rnd$
    $iv_{ZERO}$ := $iv$
    $keyword_{schema}$ := $schema$
    $iv_{ZERO}$ := newZeroFilledIV()
    $encryptedNodes$ := {}
    $keywordFrequencies$ := {}
    $derivedKeys$ := {}

// $E(T)$ := $\textrm{SSE-TP}.Encrypt([t_0, \; t_1,  ... , t_n], \; schema)$
function encrypt($triples$, $schema$):
    if ($schema$ == true)
        encryptSchemaTriples($triples$)
    else
        encryptTriples($triples$)
    return encryptContextualData()

function EncryptSchemaTriples($triples$):
    foreach $t = (s,p,o)$ in $triples$:
        encryptSchemaNode(parseNode($s$))
        encryptSchemaNode(parseNode($p$))
        encryptSchemaNode(parseNode($o$))
        
function encryptSchemaNode($node$):
    $freq$ := incrementKeywordFrequency($keyword_{schema}$)
    $trapdoor$ := E(deriveKey($keyword_{schema}$), $keyword_{schema}$, $freq$)
    $encryptedNodes$[$trapdoor$] := E($K_{RND}$, $node$)


// $E(T)$ := $\textrm{SSE-TP}.Decrypt(E(node), \; schema)$
function decrypt($ct$, $schema$):
    if ($schema$ == true)
        return D($K_{RND}$, $ct$)
    else
        return decryptNode($ct$)

function decryptContextualData($ct$):
    return D($K_{RND}$, $ct$)
    
function encryptTriples($triples$):
	foreach $t = (s,p,o)$ in $triples$:
        $freq$ := {}
        $parsed_s$ := parseNode($s$)
        $parsed_p$ := parseNode($p$)
        $parsed_o$ := parseNode($o$)
        $keyword_s$ := parseKeyword($s$)
        $keyword_p$ := parseKeyword($p$)
        $keyword_o$ := parseKeyword($o$)
        $freq$ := $freq$ :U { encryptNode($parsed_p$, "PO" || $keyword_s$) }
        $freq$ := $freq$ :U { encryptNode($parsed_o$, "PO" || $keyword_s$) }
        $freq$ := $freq$ :U { encryptNode($parsed_s$, "SO" || $keyword_p$) }
        $freq$ := $freq$ :U { encryptNode($parsed_o$, "SO" || $keyword_p$) }
        $freq$ := $freq$ :U { encryptNode($parsed_s$, "SP" || $keyword_o$) }
        $freq$ := $freq$ :U { encryptNode($parsed_p$, "SP" || $keyword_o$) }
        $freq$ := $freq$ :U { encryptNode($parsed_s$, "S" || $keyword_p$ || $keyword_o$) }
        $freq$ := $freq$ :U { encryptNode($parsed_p$, "P" || $keyword_s$ || $keyword_o$) }
        $freq$ := $freq$ :U { encryptNode($parsed_o$, "O" || $keyword_s$ || $keyword_p$) }
        encryptTriple("SPO" || $keyword_s$ || $keyword_p$ || $keyword_o$, $freq$)
 
function encryptTriplePattern($keyword$, $frequencies$): 
    $i$ := 0
    foreach $freq$ in $frequencies$:
        $trapdoor$ := E(deriveKey($keyword$), $keyword$, $i$)
        $encryptedNodes$[$trapdoor$] := E($K_{RND}$, $freq$)
        $i$++

function deriveKey($context$):
    $key$ := $derivedKeys$[$context$]
    if $key$ == :bot :
        $key$ := generateKey($K_{MASTER}$, $context$)
        $derivedKeys$[$context$] := $key$
    return $key$

function incrementKeywordFrequency($keyword$):
    $freq$ := $keywordFrequencies$[$keyword$]
    if $freq$ == :bot :
        $freq$ := 1
        $keywordFrequencies$[$keyword$] := $freq$
    else:
        $keywordFrequencies$[$keyword$] := $freq$ + 1
    return $freq$

// $trapdoor_{n,i}$ := $\textrm{SSE-TP}.Trapdoor(keyword_n, \; i)$
function generateTrapdoor($keyword$, $i$):
    return E(deriveKey($keyword$), $keyword$, $i$)
    
function generateTrapdoor($keyword$):
    return E(deriveKey($keyword$), $keyword$, $iv_{ZERO}$)
	
function generateTrapdoorAndIncrementIV($keyword$):
    return E(deriveKey($keyword$), $keyword$, incrementKeywordFrequency($keyword$))

function generateTriplePatternTrapdoors($triples$):
    $keywordPatternTrapdoors$ := {}
    foreach $t = (s,p,o)$ in $triples$:
        $keywords$ := {}
        $i$ := 0
        $keyword_s$ = parseKeyword($s$)
        $keyword_p$ = parseKeyword($p$)
        $keyword_o$ = parseKeyword($o$)
        $keyword_t$ = "SPO" || $keyword_s$ || $keyword_p$ || $keyword_o$
        $keywords$ := $keywords$ :U { "PO" || $keyword_s$ }
        $keywords$ := $keywords$ :U { "PO" || $keyword_s$ }
        $keywords$ := $keywords$ :U { "SO" || $keyword_p$ }
        $keywords$ := $keywords$ :U { "SO" || $keyword_p$ }
        $keywords$ := $keywords$ :U { "SP" || $keyword_o$ }
        $keywords$ := $keywords$ :U { "SP" || $keyword_o$ }
        $keywords$ := $keywords$ :U { "S" || $keyword_p$ || $keyword_o$ }
        $keywords$ := $keywords$ :U { "P" || $keyword_s$ || $keyword_o$ }
        $keywords$ := $keywords$ :U { "O" || $keyword_s$ || $keyword_p$ }
        foreach $keyword$ in $keywords$:
            $trapdoors$ := $keywordPatternTrapdoors$[keyword]
            if ($trapdoors$ == :bot):
                $trapdoors$ := {}
            $trapdoors$ := $trapdoors$ :U { E(deriveKey($keyword_t$), $keyword_t$, $i$) }
            $keywordPatternTrapdoors$[$keyword$] := $trapdoors$
            $i$++
    return $keywordPatternTrapdoors$
    
function setKeywordFrequencies($values$):
    foreach $keyword$ in $values$:
        $freq$ := $values$[$keyword$]
        if ($freq$ > 0)
            $keywordFrequencies$[$keyword$] := $freq$

function getRNDKey():
    return $K_{RND}$

function getEncryptedNodes():
    return $encryptedNodes$

    $encryptedNodes$ := {}

function clearFrequencies():
    $keywordFrequencies$ := {}
