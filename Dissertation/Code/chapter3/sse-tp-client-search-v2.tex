State:
    $secrets$ := {}

function search($triplestoreID$, $triplePatterns$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    $scheme$ := createEncryptionScheme()
    $scheme$.loadState($secrets$[$triplestoreID$])
    $keywords$ := [$\#triplePatterns$]
    $i$ := 0
    foreach $t$ in $triplePatterns$:
        $keyword$ := extractKeyword($triplePattern$) 
        if $keyword$ :neq :bot:
            $keywords$[$i$] := $keyword$
            $i$++
    $frequencies$ := fetchKeywordsFrequencies($triplestoreID$, $keywords$, $scheme$)
    $trapdoors$ := createLinkedList()
    for $keyword$ in $keywords$:
        for $i$ in range(0, $frequencies$[$keyword$]):
            $trapdoors$.add($scheme$.generateTrapdoorAndIncrementIV($keyword$))
    $encryptedEqTags$ := Triplestore.search($triplestoreID$, $trapdoors$)
    $encryptedNodes$ := Triplestore.search($triplestoreID$, $encryptedEqTags$)
    $results$ := {}
    foreach $node$ in $encryptedNodes$:
        if $node$ :neq :bot:
            $results$ := $results$ :U { $scheme$.decrypt($node$, false) }
    return $results$
