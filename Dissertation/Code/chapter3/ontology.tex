State:
    $subClasses$ := {}
    $equivalentClasses$ := {}
    $classRestrictions$ := {}
    $intersectionClasses$ := {}
    $intersectionsWhereClassIsOperand$ := {}
    $subProperties$ := {}
    $equivalentProperties$ := {}
    $inverseProperties$ := {}
    $symmetricProperties$ := {}
    $transitiveProperties$ := {}
    $transitivityDepth$ := :bot
    $expansionDepth$ := :bot

function execInference($schema$, $e$, $t$):
    $expansionDepth$ := $e$
    $transitivityDepth$ := $t$
    $tbox$ = JENA$_{ModelFactory}$.createOntologyModel(JENA$_{spec}$.OWL_MEM)
    JENA$_{GraphUtil}$.add($tbox$.getGraph(), $schema$)
    $model$ = JENA$_{ModelFactory}$.createOntologyModel(JENA$_{spec}$.OWL_MEM_TRANS_INF, $tbox$)
    execClassInference($model$.listClasses())
    execPropertyInference($model$.listAllOntProperties())

function execClassInference($classes$, $restrictions$, $intersections$):
    foreach $class = (subclasses, \; equivalentClasses)$ in $classes$:
        $subClasses$[convertToNode($class$)] := $subclasses$
        $equivalentClasses$[convertToNode($class$)] := $equivalentClasses$
    foreach $restriction = (type, \; property, \; value)$ in $restrictions$:
        $type \in \{$"owl:hasValue", "owl:isSomeValuesFrom", "owl:allValuesFrom"$\}$:
            $classRestrictions$[convertToNode($restriction$)] := $restriction$
    foreach $intersection = (id, operands)$ in $intersections$: // Intersections are anonymous classes with a set of other classes (operands)
        foreach $operand$ in $operands$:
            $intersectionsDirectSuperclasses$ := $intersectionsWhereClassIsOperand$[convertToNode($operand$)]
            if $intersectionsDirectSuperclasses$ == :bot:
                $intersectionsDirectSuperclasses$ := {}
            $intersectionsDirectSuperclasses$ := $intersectionsDirectSuperclasses$ :U $intersection$
            $intersectionsWhereClassIsOperand$[convertToNode($operand$)] := $intersectionsDirectSuperclasses$
        $intersectionClasses$[convertToNode($intersection$)] := $operands$

function execPropertyInference($properties$):
    foreach $p = (type, \; value, \; subproperties, \; equivalentProperties, \; inverseProperties)$ in $properties$:
        if $type$ == "owl:SymmetricProperty":
            $symmetricProperties$ := $symmetricProperties$ :U { $p$ }
        elif $type$ == "owl:TransitiveProperty":
            $transitiveProperties$ := $transitiveProperties$ :U { $p$ }
        $subProperties$[convertToNode($p$)] := $subproperties$
        $equivalentProperties$[convertToNode($p$)] := $equivalentProperties$
        $inverseProperties$[convertToNode($p$)] := $inverseProperties$
    
function getEquivalentClasses($node$):
    $classes$ := $equivalentClasses$[$node$]
    return $classes$ == :bot:
        return {}
    return $classes$

function getSubClasses($node$):
    $classes$ := $subClasses$[$node$]
    return $classes$ == :bot:
        return {}
    return $classes$

function getRestriction($node$):
    return $classRestrictions$[node]

function getIntersection($node$):
    $classes$ := $intersectionClasses$[$node$]
    return $classes$ == :bot:
        return {}
    return $classes$

function getIntersectionWhereClassIsOperand($node$):
    $classes$ := $intersectionsWhereClassIsOperand$[$node$]
    return $classes$ == :bot:
        return {}
    return $classes$

function getSubProperties($node$): 
    $properties$ := $subProperties$[$node$]
    return $properties$ == :bot:
        return {}
    return $properties$

function getEquivalentProperties($node$):
    $properties$ := $equivalentProperties$[$node$]
    if $properties$ == :bot:
        return {}
    return $properties$

function getInverseOf($node$):
    $properties$ := $inverseProperties$[$node$]
    if $properties$ == :bot:
        return {}
    return $properties$
    
function isSymmetric($node$): 
    return $node \in symmetricProperties$

function isTransitive(node):
    return $node \in transitiveProperties$

function getTransitivityDepth():
    return $transitivityDepth$

function getMaximumExpansionDepth():
    return $expansionDepth$
