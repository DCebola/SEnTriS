State:
    $secrets$ := {}

function fetchKeywordsFrequencies($triplestoreID$, $keywords$, $scheme$):
    $\textrm{\emph{f}}requencies$ := {}
    $trapdoors$ := [$\#keywords$]
    for $keyword$ in $keywords$:
        $trapdoors$.add($scheme$.generateTrapdoor($keyword$))
    $encryptedFrequencies$ := Triplestore.search($triplestoreID$, $trapdoors$)
    for $i$ in range(0, $\#encryptedFrequencies$) 
        $\textrm{\emph{f}}requency$ := $encryptedFrequencies$.get($i$)
        if $\textrm{\emph{f}}requency$ :neq :bot: 
            $\textrm{\emph{f}}requencies$[$keywords$[$i$]] := $scheme$.decryptContextualData($\textrm{\emph{f}}requency$)
        else
            $\textrm{\emph{f}}requencies$[$keywords$[$i$]] := 0
    return $\textrm{\emph{f}}requencies$

// $\{[node_0, ..., node_n], \; \bot\} \leftarrow \textrm{SSE-TP}.SearchTriplestore(triplestoreID, \; triplePatterns)$
function search($triplestoreID$, $triplePatterns$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    $scheme$ := createEncryptionScheme()
    $scheme$.ReInitialize($secrets$[$triplestoreID$])
    $keywords$ := [$\#triplePatterns$]
    $i$ := 0
    foreach $t$ in $triplePatterns$:
        $keyword$ := extractKeyword($triplePattern$) // Parses the pattern and, based on its type, generates a keyword like the encryption scheme would. Outputs :bot  if pattern not supported.
        if $keyword$ :neq :bot:
            $keywords$[$i$] := $keyword$
            $i$++
    $\textrm{\emph{f}}requencies$ := fetchKeywordsFrequencies($triplestoreID$, $keywords$, $scheme$)
    $trapdoors$ := createLinkedList()
    for $keyword$ in $keywords$:
        for $i$ in range(0, $\textrm{\emph{f}}requencies$[$keyword$]):
            $trapdoors$.add($scheme$.generateTrapdoorAndIncrementIV($keyword$))
    $encryptedNodes$ := Triplestore.search($triplestoreID$, $trapdoors$)
    $results$ := {}
    foreach $node$ in $encryptedNodes$:
        if $node$ :neq :bot:
            $results$ := $results$ :U { $scheme$.decrypt($node$, false) }
    return $results$    

// $\{[t_0, t_1, ..., t_n], \; \bot\} \leftarrow \textrm{SSE-TP}.SearchSchema(triplestoreID)$
function searchSchema($triplestoreID$):
    if $secrets$[$triplestoreID$] == :bot
        return :bot
    $scheme$ := createEncryptionScheme()
    $scheme$.ReInitialize($secrets$[$triplestoreID$])
    $keyword$ := $scheme$.getSchemaKeyword()
    $\textrm{\emph{f}}requencies$ := fetchKeywordsFrequencies($triplestoreID$, [$keyword$], $scheme$)
    $trapdoors$ := createLinkedList()
    for $i$ in range(0, $\textrm{\emph{f}}requencies$[$keyword$]):
        $trapdoors$.add($scheme$.generateTrapdoorAndIncrementIV($keyword$))
    $encryptedNodes$ := Triplestore.search($triplestoreID$, $trapdoors$)
    $schema$ := {}
    $i$ := 0
    while $i < \textrm{\emph{f}}requencies$[$keyword$]:
        $s$ := $scheme$.decrypt($encryptedNodes$.get($i$), true)
        $p$ := $scheme$.decrypt($encryptedNodes$.get($i$ + 1), true)
        $o$ := $scheme$.decrypt($encryptedNodes$.get($i$ + 2), true)
        $schema$ := $schema$ :U  { $(s, \; p, \; o)$ } 
        $i$ := $i$ + 3
    return $schema$
