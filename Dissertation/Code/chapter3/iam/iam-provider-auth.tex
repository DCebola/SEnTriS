State:
    $storageEngine$ := createStorageEngine()
    $locksClient$ := createLocksClient()

function createSession($username$, $password$):
    $cookie$ := :bot
    if checkPassword($username$, $password$):
        $cookie$ := $storageEngine$.cacheSession($username$);
    return $cookie$

function saveUser($username$, $password$):
    $lockID$ := $locksClient$.acquireUserLock($username$)
    if $storageEngine$.getUser(username) :neq :bot:
        $lockID$ := $locksClient$.releaseUserLock($username$)
        return :bot
    $hash$ := hash($password$) // Generates a hash using a one-way key derivation function, and a randomly generated salt
    $storageEngine$.saveUser($username$, $hash$, "BASIC")
    $locksClient$.releaseUserLock($username$, $lockID$)
    return "SUCCESS"

function deleteUser($cookie$, $username$):
    if authCheck($cookie$, $username$):
    $lockID$ := $locksClient$.acquireUserLock($username$)
    if $storageEngine$.checkIfOwnsAny($username$): // Triplestores owned by the user must be deleted beforehand
        $locksClient$.releaseUserLock($username$, $lockID$)
        return :bot
    $storageEngine$.deleteUser($username$)
    $locksClient$.deleteAllUserLocks($username$)
    $locksClient$.releaseUserLock($username$, $lockID$)
    return "SUCCESS"

function checkPassword($username$, $password$):
    $hash$ := $storageEngine$.getPassword($username$)
    if $hash$ := :bot
        $salt$ := extractSalt($hash$)
        $hash_2$ := hash($password$, $salt$) // Generates a hash a one-way key derivation function, and the extracted salt
        return $hash_2$ = $hash$ // We will assume a secure implementation of the comparison
