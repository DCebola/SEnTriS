State:
    $storageEngine$ := createStorageEngine()
    $locksClient$ := createLocksClient()

function createAccessToken($cookie$, $triplestoreID$, $target$):
    try: 
        authCheck($cookie$, $target$)
        if $storageEngine$.triplestoreAccessListExists($triplestoreID$) = false:
            return :bot
        $role$ := $storageEngine$.getRole($target$)
        if $role$ :neq "ADMIN" and $storageEngine$.checkIfUserHasReadAccess($target$, $triplestoreID$) = false and $storageEngine$.checkIfUserHasWriteAccess($target$, $triplestoreID$) = false and $storageEngine$.checkIfOwns($target$, $triplestoreID$) = false:
            return :bot
        return $storageEngine$.saveToken($target$, $cookie$.getValue(), $triplestoreID$)
    catch Exception:
        return :bot

function deleteAccessToken($cookie$, $triplestoreID$, $headers$):
    try: 
        $(tokenID, \; username, \; sessionID, \; triplestoreID_{token}, \; lockID)$ := fetchAccessToken($headers$)
        sessionCheck($sessionID$, $username$)
        if $storageEngine$.triplestoreAccessListExists($triplestoreID$) = false:
            return :bot
        if $triplestore_{token}$ :neq $triplestoreID$:
            return :bot
        $storageEngine$.deleteAccessToken($tokenID$, $lockID$, $username$, $triplestoreID_{token}$)
        return "SUCCESS"
    catch Exception: 
        return :bot

function acquireTriplestoreLock($cookie$, $triplestoreID$, $headers$) 
    try: 
        $(tokenID, \; username, \; sessionID, \; triplestoreID_{token}, \; lockID)$ := fetchAccessToken($headers$) 
        authCheck($cookie$, $username$)
        if $cookie$.getValue() :neq $sessionID$: 
            return :bot
        if $storageEngine$.triplestoreAccessListExists($triplestoreID$) = false:
            return :bot
        if $triplestore_{token}$ :neq $triplestoreID$:
            return :bot
        $role$ := $storageEngine$.getRole($username$)
        if $role$ :neq "ADMIN" and $storageEngine$.checkIfUserHasWriteAccess($username$, $triplestoreID$) = false and $storageEngine$.checkIfOwns($username$, $triplestoreID$) = false:
            return :bot
        $lockID$ := $locksClient$.acquireTriplestoreLock($username$, $triplestoreID$)
        $storageEngine$.addLockToToken($tokenID$, $lockID$)
        return $lockID$
    catch Exception: 
        return :bot

function releaseTriplestoreLock($cookie$, $triplestoreID$, $headers$) 
    try: 
        $(tokenID, \; username, \; sessionID, \; triplestoreID_{token}, \; lockID)$ := fetchAccessToken($headers$) 
        authCheck($cookie$, $username$)
        if $cookie$.getValue() :neq $sessionID$: 
            return :bot
        if $storageEngine$.triplestoreAccessListExists($triplestoreID$) = false:
            return :bot
        if $triplestore_{token}$ :neq $triplestoreID$:
            return :bot
        if $lockID$ = :bot:
            return :bot
        $storageEngine$.deleteLockFromToken($tokenID$)
        $locksClient$.releaseTriplestoreLock($username$, $triplestoreID$, $lockID$)
        return "SUCCESS"
     catch Exception: 
        return :bot

function fetchAccessToken($headers$):
    $tokenID$ := extractAccessToken($headers$) // Extracts the value of the access token from the headers
    if $tokenID$ = :bot
        throw TokenException
    $token$ := $storageEngine$.getToken($tokenID$)
    if $token$ = :bot: // Expired or invalid
        throw TokenException
    return $(tokenID, \; token.username, \; token.sessionID, \; token.triplestoreID, \; token.lockID)$
