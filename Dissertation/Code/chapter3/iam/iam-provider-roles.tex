State:
    $storageEngine$ := createStorageEngine()
    $locksClient$ := createLocksClient()

funciton issueGrantRoleRequest($cookie$, $username$, $issuer$, $role$) 
    try:
        authCheck($cookie$, $issuer$)
        if $storageEngine$.userExists($username$) = false:
            return :bot
        $role{issuer}$ := $storageEngine$.getRole($issuer$)
        if $storageEngine$.getRole($username$) = $role$:
            return :bot
        if $role{issuer}$ = "BASIC" or $role{issuer}$ = "PRIVILEGED":
            if $username$ = $issuer$:
                $storageEngine$.saveRoleRequest($username$, $role$)
                return "SUCCESS"
            else:
                return :bot
        else:
            $lockID$ := $locksClient$.acquireUserLock($username$)
            $storageEngine$.setRole($username$, $role$)
            $locksClient$.releaseUserLock($username$, $lockID$)
            return "SUCCESS"
    catch Exception: 
        return :bot

function getPendingRoleRequests($cookie$, $username$) 
    try:
        authCheck($cookie$, $username$)
        $role$ := $storageEngine$.getRole($username$)
        if $role$ :neq "ADMIN":
            return :bot
        return $storageEngine$.getPendingRoleRequests()
    catch Exception: 
        return :bot

function processRoleRequest($cookie$, $username$, $requestID$, $target$, $decision$): 
    try:
        authCheck($cookie$, $username$)
        $role{issuer}$ := $storageEngine$.getRole($username$)
        if $role{issuer}$ :neq "ADMIN":
            return :bot
        if $storageEngine$.userExists($target$) = false:
            return :bot
        $lockID$ := $locksClient$.acquireUserLock($target$)
        $(username_{request}, \; role)$ := $storageEngine$.getPendingRoleRequest($requestID$)
        if $request$ = :bot: 
            $locksClient$.releaseUserLock($target$, $lockID$)
            return :bot
        if $target$ :neq $username_{request}$:
            $locksClient$.releaseUserLock($target$, $lockID$)
            return :bot
        if $decision$ = true:
            if $storageEngine$.getRole($target$) = $role$:
                $storageEngine$.deleteRoleRequest($requestID$)
                $locksClient$.releaseUserLock($target$, $lockID$)
                return :bot
            $storageEngine$.setRole($target$, $role$)
        $storageEngine$.deleteRoleRequest($requestID$)
        $locksClient$.releaseUserLock($target$, $lockID$)
        return "SUCCESS"
    catch Exception: 
        return :bot
