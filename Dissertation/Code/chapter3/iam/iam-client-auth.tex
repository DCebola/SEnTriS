function registerUser($username$, $password$):
    return IAM.saveUser($username$, $password$)

function auth($username$, $password$):
    $response$ := IAM.createSession($username$, $password$)
    if $response$ :neq :bot:
        return extractCookie($response$)
    return $response$

function unregisterUser($cookie$, $username$):
    return IAM.deleteUser($cookie$, $username$)
