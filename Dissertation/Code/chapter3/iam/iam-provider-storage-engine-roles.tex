State:
    $SESSION\_LIFETIME$ := fetchConfig($SESSION\_LIFETIME$)
    $roles$ := {}
    $passwords$ := {}
    $sessions$ := {}
    $owned$ := {}
    $roleRequests$ := {}

function setRole($username$, $role$):
    $roles$[$username$] := $role$

function deleteUser($username$)
    $roles$[$username$] := :bot
    $passwords$[$username$] := :bot
    $owned$[$username$] := :bot
    $sessions$[$username$] := :bot

function expireSession($sessionID$, $username$):
    if $sessions$[$username$] = $sessionID$:
        $sessions$[$username$] := :bot

function saveRoleRequest($username$, $role$):
    $requestID$ := generateID()
    $roleRequests$[$requestID$] := ($username$, $role$)

function getPendingRoleRequest($requestID$):
    return $roleRequests$[$requestID$]

function getPendingRoleRequests():
    $pendingRequests$ := {}
    foreach $(requestID, \; (username, \; role))$ in $roleRequests$:
        $pendingRequests$ := $pendingRequests$ :U { $(requestID, \; username, \; role)$ }
    return $pendingRequests$

function deleteRoleRequest($requestID$):
    $roleRequests$[$requestID$] := :bot
