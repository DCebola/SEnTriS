State:
    $operations_{parsed}$ := {}
    $upload_{template}$ := {}
    $delete_{template}$ := {}
    $construct_{template}$ := {}
    $queryType$ := $:bot$
    $plan$ := ($jobs$ = {}, $execOrder$ = createLinkedList(), $vars$ = {})
    $variables$ := {}

function getQueryType():
    return $queryType$

function setQueryType($type$):
    $queryType$ := $type$

function setConstructTemplate($triples$) 
    $construct_{template}$ := $triples$

function getConstructTemplate: 
    return $construct_{template}$

function getUploadTemplate:
    return $upload_{template}$

function getDeleteTemplate:
    return $delete_{template}$

function pushJob($job$):
    $jobID$ := $job$.getID()
    $plan$.$jobs$[$jobID$] := $job$
    $plan$.$execOrder$.addLast($jobID$)

function generatePlan($op$) 
    JENA.$_{OpWalker}$.walk($op$, $this$) # the operation walker will call visit while there are operations which have not been visited, using a visitor (the query planner is a JENA$_{Visitor}$)
    if $\#plan.vars$ == 0:
        $plan.vars$ := $variables$
    return $plan$

function generatePlan($op$, JENA$_{algebraGenerator}$):
    if $op$ is UpdateDataInsert:
        visitUpdate($op$)
    elif $op$ is UpdateDataDelete: 
        visitUpdate($op$)
    elif $op$ is UpdateModify:
        visitUpdate($op$, JENA$_{algebraGenerator}$)
    elif $op$ is UpdateDeleteWhere:
        visitUpdate($op$)
    else: 
        throw NotImplementedException()
    if $\#plan.vars$ == 0:
        $plan.vars$ := $variables$
    return $plan$

function visit(op) 
    if $op$ is OpBGP:
        generateSearchJobs($op$)
    elif $op$ is OpTriple:
        generateSearchJobs($op$.asBGP())
    elif $op$ is OpJoin:
        generateJoinJob($op$)
    elif $op$ is OpLeftJoin: 
        generateOptionalJob($op$)
    elif $op$ is OpMinus:
        generateMinusJob($op$)
    elif $op$ is OpUnion: 
        generateUnionJob($op$)
    elif $op$ is OpModifier:
        if $op$ is OpDistinctReduced: 
            generateDistinctJob($op$)
        elif $op$ is OpOrder: 
            generateOrderByJob($op$)
        elif $op$ is OpProject: 
            generateProjectJob($op$)
        elif $op$ is OpSlice: 
            generateSliceJob($op$)
        else 
            throw NotImplementedException()
    else:
        throw NotImplementedException()

function visitUpdate($op_{UpdateDataInsert}$) 
    setQueryType("INSERT_DATA")
    foreach $quad = (graph, \; triple)$ in $op_{UpdateDataInsert}$.getQuads(): # we only support a single default graph, hence quads can be converted to triples
        $upload_{template}$ := $upload_{template}$ :U { triple }

function visitUpdate($op_{UpdateDataDelete}$) 
    setQueryType("DELETE_DATA")
    foreach $quad = (graph, \; triple)$ in $op_{UpdateDataDelete}$.getQuads():
        $delete_{template}$ := $delete_{template}$ :U { triple }

function visitUpdate($op_{UpdateDeleteWhere}$):
    setQueryType("MODIFY")
    $bgp$ := {}
    foreach $quad = (graph, \; triple)$ in $op_{UpdateDeleteWhere}$.getQuads():
        $bgp$ := $bgp$ :U {$triple$}
        $delete_{template}$ := $delete_{template}$ :U { triple }
    JENA.$_{OpWalker}$.walk(JENA.createOpBGP($bgp$), $this$)

function visitUpdate($op_{UpdateModify}$, JENA$_{algebraGenerator}$) 
    setQueryType("MODIFY")
    if $op_{UpdateModify}$.hasInsertClause():
        foreach $quad = (graph, \; triple)$ in $op_{UpdateDataInsert}$.getInsertQuads():
            $upload_{template}$ := $upload_{template}$ :U { triple }
    if $op_{UpdateModify}$.hasDeleteClause():
        foreach $quad = (graph, \; triple)$ in $op_{UpdateDeleteWhere}$.getDeleteQuads():
            $delete_{template}$ := $delete_{template}$ :U { triple }
        JENA.$_{OpWalker}$.walk(JENA$_{algebraGenerator}$.compile($op_{UpdateModify}$.getWherePattern()), $this$)
            
function generateSearchJobs($op_{OpBGP}$):
    $patterns$ = $op_{OpBGP}$.getPattern()
    $jobIDs$ = {}
    $jobs$ = {}
    $jobVars$ = {}
    foreach $t = (s, p, o)$ in $patterns$:
        $jobID$ = pushSearch($s$, $p$, $o$, $jobs$, $jobIDs$)
        $job$ = $jobs$[$jobID$]
        $vars_{job}$ := extractVars($job$)
        $variables$ := $variables$ :U $vars_{job}$
        $jobVars$[$jobID$] := $vars_{job}$
    if $\#patterns$ > 1:
        generateRandomJoinPipeline($op_{OpBGP}$, $jobVars$)
    else:
        $operations_{parsed}$.put($op_{OpBGP}$, $jobID$)
    
function pushSearch($s$, $p$, $o$, $jobs$, $jobIDs$):
    $searchPattern$ := parsePattern($s$, $p$, $o$)
    $jobID$ = $jobIDs$.get($searchPattern$)
    if $jobID$ == $:bot$: 
        $jobID$ := generateID()
        $job_{SEARCH}$ := createSearchJob($jobID$, extractVariablesPattern($s$, $p$, $o$), $s$, $p$, $o$)
        $jobs$[$jobID$] := $job_{SEARCH}$
        $jobIDs$[$searchPattern$] := $jobID$
        pushJob($job_{SEARCH}$)
    return $jobID$

function generateAdjacencyMatrix($jobVars$) 
    $graph$ := {}
    $jobs$ := {}
    $vars$ := {}
    foreach $jobID_1$ in $jobVars$.getKeys(): 
        $jobs$ := $jobs$ :U { $jobID_1$ }
        $v_1$ := $jobVars$[$jobID_1$]
        $vars$ := $vars$ :U $\;v_1$
        $edges$ := {}
        foreach $jobID_2$ in $jobVars$.getKeys():
            $v_2$ := $jobVars$[$jobID_2$]
            if $jobID_1$ :neq $jobID_2$
                foreach $var$ \; in \; $v_1$:
                    if $v \in v_2$:
                        $edges$ := $edges$ :U { $JobID_2$ }
                        break
        $graph$[$jobID_1$] := $edges$
    return ($graph$, $jobs$, $vars$)


function bfsSearch($root$, $graph$, $path$, $depth$) 
    $queue$ = createQueue()
    $path$.add($root$)
    $queue$.push($root$)
    while $\#path$ < $depth$: 
        $next$ := $queue$.poll()
        foreach $neighbour$ in $graph$[$next$]:
            if $neighbour \; \notin \; path$:
                $path$.add($neighbour$)
                $queue$.push($neighbour$)
        if $\#queue$ == 0:
            return  

function getJobID($op$) 
    $jobID$ = $operations_{parsed}$[$op$]
    if $jobID$ == :bot:
        throw QueryBuildException()
    return $jobID$

function generateProjectJob($op_{OpProject}$) 
    $jobID$ := generateID()
    $variables$ := $op$.getVars()
    $plan$.$vars$ := $variables$
    pushJob(createProjectJob($jobID$, getJobID($op$.getSubOp()), $variables$))
    $operations_{parsed}$[$op$] := $jobID$
