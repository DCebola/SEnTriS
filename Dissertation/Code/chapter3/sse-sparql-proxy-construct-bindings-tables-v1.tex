State:
    $TIMEOUT$ := fetchConfig($TIMEOUT$)
    $bindings$ := {}

function search($K_{RND}$, $variables$, $searches$):
    $table$ := createBindingsTable()
    $table$.setVariables($variables$)
    $searchResults$ := {}
    foreach $variable$ in $variables$:
        $searchResults$[$variable$] := $bindings$[$searches$[$variable$]]
    for $i$ in range(0, $\#searchResults$[$variables$[0]]):
        $patternID$ := generateID()
        foreach $variable$ in $variables$:
            $table$.add($patternID$, $variable$, D($K_{RND}$, $searchResults$[$variable$][$i$]))
    return $table$
