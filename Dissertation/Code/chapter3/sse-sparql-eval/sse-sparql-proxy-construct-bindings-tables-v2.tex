        for $i$ in range(0, $\#bindings_{var}$):
            $searchResults$[$variable$][$i$] := $searchBindings_{var}$
    $groupedEqTags$ := {}
    foreach $variable$ in $variables$:
        $groupedEqTags$[$variable$] := {}
    for $i$ in range(0, $\#searchResults$[$variables$[0]]):
        $patternID$ := generateID()
        foreach $variable$ in $variables$:
            $eqTag$ := $searchResults$[$variable$][$i$]
            $binding$ := removeRNDLayer($K_{DGK_{priv}}$, $eqTag$)

                $eqTags$[$execID$] := {}
            $eqTags$[$execID$][$binding$] := $eqTag$

    return $table$
