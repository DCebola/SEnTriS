State:
    $TIMEOUT$ := fetchConfig($TIMEOUT$)
    $triplestores$ := {}
    $uploads$ := {}
    $deletions$ := {}

function search($triplestoreID$, $mask$, $trapdoors$):
    $result$ := createArrayList($\#trapdoors$)
    for $i$ in range(0, $\#trapdoor$):
        $binding$ := $triplestores$[$triplestoreID$][$trapdoors$[$i$]]
        if $binding$ :neq :bot
            $result$.add($binding \; * \; mask$)
    return $result$
