    if $\#plan.vars$ == 0:
        $plan.vars$ := $ob\textrm{\emph{f}}uscationMap$.getKeys()
    return $plan$

function obfuscate(var):
    $var_{ob\textrm{\emph{f}}uscated}$ := $ob\textrm{\emph{f}}uscationMap$[$var$]
    if $var_{ob\textrm{\emph{f}}uscated}$ == :bot:
        $var_{ob\textrm{\emph{f}}uscated}$ := generateID()
        $ob\textrm{\emph{f}}uscationMap$[var] := $var_{ob\textrm{\emph{f}}uscated}$
        $deobfuscationMap$[$var_{ob\textrm{\emph{f}}uscated}$] := $var$
    return $var_{ob\textrm{\emph{f}}uscated}$

function generateSecureSearchJob($var$, $node_1$, $node_2$, $pattern$):
    $searches$ := {}
    $var$ := obfuscate($var$)
    $jobID$ := generateID()
    $keyword$ := $pattern$ || parseKeyword($node_1$) || parseKeyword($node_2$)
    $searches$[$var$] := $keyword$
    $keywords$ := $keywords$ :U { $keyword$ }
    $searchJobsIDs$ := $searchJobsIDs$ :U { $jobID$ }
    return createSecureSearchJob($jobID$, [$var$], $searches$)

function generateSecureSearchJob($var_1$, $var_2$, $node$, $pattern$):
    $searches$ := {}
    $jobID$ := generateID()
    $keyword$ := $pattern$ || parseKeyword($node$)
    $keywords$ := $keywords$ :U { $keyword$ }
    $searchJobsIDs$ := $searchJobsIDs$ :U { $jobID$ }
    $var_1$ = obfuscate($var_1$);
    $var_2$ = obfuscate($var_2$);
    $searches$[$var_1$] := $keyword$
    $searches$[$var_2$] := $keyword$
    return createSecureSearchJob($jobID$, [$var_1$, $var_2$], $searches$);

function pushSecureSearch($s$, $p$, $o$, $jobs$, $jobIDs$)
    $searchPattern$ := parsePattern($s$, $p$, $o$)
    $jobID$ = $jobIDs$.get($searchPattern$)
    if $jobID$ == :bot: 
        $jobID$ := generateID()
        $pattern$ := extractVariablesPattern($s$, $p$, $o$)
        $job_{SEC\_SEARCH}$ := :bot
        switch ($pattern$):
            case "S": $job_{SEC\_SEARCH}$ := generateSecureSearchJob($s$, $p$, $o$, $pattern$)
            case "P": $job_{SEC\_SEARCH}$ := generateSecureSearchJob($p$, $s$, $o$, $pattern$)
            case "O": $job_{SEC\_SEARCH}$ := generateSecureSearchJob($o$, $s$, $p$, $pattern$)
            case "SP": $job_{SEC\_SEARCH}$ := generateSecureSearchJob($s$, $p$, $o$, $pattern$)
            case "SO": $job_{SEC\_SEARCH}$ := generateSecureSearchJob($s$, $o$, $p$, $pattern$)
            case "PO": $job_{SEC\_SEARCH}$ := generateSecureSearchJob($p$, $o$, $s$, $pattern$)
            default: throw NotImplementedException()
        $jobs$[$jobID$] := $job_{SEC\_SEARCH}$
        $jobIDs$[$searchPattern$] := $jobID$
        pushJob($job_{SEARCH}$)
    return $jobID$
 
function generateSearchJobs($op_{OpBGP}$):
    $patterns$ = $op_{OpBGP}$.getPattern()
    $jobIDs$ = {}
    $jobs$ = {}
    $jobVars$ = {}
    foreach $t = (s, p, o)$ in $patterns$:
        $jobID$ = pushSecureSearch($s$, $p$, $o$, $jobs$, $jobIDs$)
        if $ontology$ :neq :bot:
            if ($p$ == "rdf:type") and ($o$ is not VARIABLE):
                $jobID$ = expandClass($jobID$, $s$, $o$, 0, $jobs$, $jobIDs$)
            elif $p$ is not VARIABLE:
                $jobID$ = expandProperty($jobID$, $s$, $p$, $o$, 0, $jobs$, $jobIDs$, true, true)
        $job$ := $jobs$[$jobID$]
        if $job$ is SEC_SEARCH:
            $vars_{job}$ := $job$.getVariables()
            $variables$ := $variables$ :U $vars_{job}$
            $jobVars$[$jobID$] := $vars_{job}$
        elif $job$ is BINARY:
            $jobVars$[$jobID$] := extractVarsOfBinaryJob($jobs$, $job$)
    if $\#patterns$ > 1:
        generateRandomJoinPipeline($op_{OpBGP}$, $jobVars$)
    else:
        $operations_{parsed}$.put($op_{OpBGP}$, $jobID$)

function expandClass($jobID$, $s$, $o$, $depth$, $jobs$, $jobsIDs$):
    if $depth$ == $ontology$.getMaximumExpansionDepth():
        return $jobID$
    foreach $expandtionType$ in shuffle($classExpansionOrder$):
        if $expandtionType$ == "SUBCLASS":
            $subclasses$ := $ontology$.getSubClasses($o$)
            $jobID$ := expandClassDisjunction($subclasses$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
        elif $expandtionType$ == "EQUIVALENT_CLASS":
            $equivalentClasses$ := $ontology$.getEquivalentClasses($o$)
            $jobID$ := expandClassDisjunction($equivalentClasses$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
        elif $expandtionType$ == "RESTRICTION":
            $restriction$ := $ontology$.getRestriction($o$)
            $jobID$ := expandRestriction($restriction$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
        elif $expandtionType$ == "INTERSECTION_OPERAND":
            $intersectionClasses$ := $ontology$.getIntersectionWhereClassIsOperand($o$)   
            $jobID$ := expandClassDisjunction($intersectionClasses$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
        elif $expandtionType$ == "INTERSECTION":
            $intersection$ := $ontology$.getIntersection($o$)
            if $\#intersection$ == 0:
                $jobID$ := expandClassConjunction($intersection$, $jobID$, $s$, $depth$, $jobs$, $jobsIDs$)
    return $jobID$

function expandProperty($jobID$, $s$, $p$, $o$, $depth$, $jobs$, $jobsIDs$, $symmetric$, $inverseO\textrm{\emph{f}}$):
    if $depth$ == $ontology$.getMaximumExpansionDepth():
        return $jobID$
    foreach $expandtionType$ in shuffle($classExpansionOrder$):
        if $expandtionType$ == "SYMMETRIC":
            if $symmetric$ == true and $ontology$.isSymmetric($p$):
                $jobID$ := pushUnion($jobID$, pushSearch($o$, $p$, $s$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
                $jobID$ := expandProperty($jobID$, $o$, $p$, $s$, $depth$ + 1, $jobs$, $jobsIDs$, false, $inverseO\textrm{\emph{f}}$)
        elif $expandtionType$ == "TRANSITIVE":
            if $ontology$.isTransitive($p$):
                $var$ := createVariable(parse(p) || 0)
                $le\textrm{\emph{f}}t$ := pushSearch($s$, $p$, $var$, $jobs$, $jobsIDs$)
                $join$ := pushJoin($le\textrm{\emph{f}}t$, pushSearch($var$, $p$, $o$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
                $jobID$ := pushUnion($jobID$, $join$, $jobs$, $jobsIDs$)
                for $i$ in range(0, $ontology$.getTransitivityDepth())
                    for $j$ in range(0, $i$ + 1):
                        if $j$ == $i$: 
                            $var_{next}$ = createVariable(parse(p) || $j$)
                            $le\textrm{\emph{f}}t$ := pushJoin($le\textrm{\emph{f}}t$, pushSearch($var$, $p$, $var_{next}$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
                            $var$ = $var_{next}$
                    $join$ := pushJoin($le\textrm{\emph{f}}t$, pushSearch($var$, $p$, $o$, $jobs$, $jobsIDs$), $jobs$, $jobsIDs$)
                    $jobID$ := pushUnion($jobID$, $join$, $jobs$, $jobsIDs$)
        elif $expandtionType$ == "SUBPROPERTY":
            $subproperties$ := $ontology$.getSubProperties($p$)
            $jobID$ := expandProperties($subproperties$, $jobID$, $s$, $o$, $depth$, $jobs$, $jobsIDs$, $symmetric$, $inverseO\textrm{\emph{f}}$)
        elif $expandtionType$ == "EQUIVALENT_PROPERTY":
            $equivalentProperties$ := $ontology$.getEquivalentProperties($p$)
            $jobID$ := expandProperties($equivalentProperties$, $jobID$, $s$, $o$, $depth$, $jobs$, $jobsIDs$, $symmetric$, $inverseO\textrm{\emph{f}}$)
        elif $expandtionType$ == "INVERSE_OF":
            if $inverseO\textrm{\emph{f}}$ == true:
                $inverseProperty$ := $ontology$.getInverseOf($p$)
                $jobID$ := expandProperties($inverseProperty$, $jobID$, $o$, $s$, $depth$, $jobs$, $jobsIDs$, $symmetric$, false)
    return $jobID$
