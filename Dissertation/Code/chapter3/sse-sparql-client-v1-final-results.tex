function generateFinalResult($triplestoreID$, $queryType$, $variables$, $sparqlResult$, $construct_{template}$, $upload_{template}$, $delete_{template}$)
    
    //Decrypt Bindings & Deobfuscate Variables

    if $queryType$ == "SELECT":
        return generateSELECTResults($variables$, $sparqlResult$)
    elif $queryType$ == "CONSTRUCT":
        return generateCONSTRUCTResults($construct_{template}$, $sparqlResult.bindings$)
    elif $queryType$ == "ASK":
        return generateASKResults($sparqlResult$.$bindings$)
    elif $queryType$ == "DESCRIBE":
        return generateDESCRIBEResults($variables$, $sparqlResult$)
    else:
        return applyModifications($triplestoreID$, $sparqlResult$.$bindings$, $upload_{template}$, $delete_{template}$)   

function applyModifications($triplestoreID$, $bindings$, $upload_{template}$, $delete_{template}$):
    $triples_{upload}$ := {}
    $triples_{delete}$ := {}
    if $queryType$ == "MODIFY":
        $triples_{upload}$ := generateTriplesFromBindings($upload_{template}$, $bindings$) 
    $triples_{delete}$ := generateTriplesFromBindings($delete_{template}$, $bindings$)
    else if (queryType == INSERT_DATA)
        $triples_{upload}$ := $upload_{template}$
    else if (queryType == DELETE_DATA)
        $triples_{delete}$ := $delete_{template}$
    return updateTriplestore($triplestoreID$, $triples_{delete}$, $triples_{upload}$)
