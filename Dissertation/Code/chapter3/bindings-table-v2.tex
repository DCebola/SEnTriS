function equals($K_{DGK_{eq}} = (p, \; vp, \; n, \; u)$, $ct_1$, $ct_2$):
    $\Delta$ := $ct_1 \; * \; ct_2\;^{(u - 1)} \; \textrm{mod} \; n$
    if $\Delta^{vp} \; \textrm{mod} \; p$ = 1:
        return true
    return false

function join($other$, $K_{DGK_{eq}}$):
    $vars_{mutual}$ := copy(getVariables())
    $vars_{mutual}$.retain($other$.getVariables())
    return join($vars_{mutual}$, $this$, $other$, "INNER", $K_{DGK_{eq}}$)

function leftOuterJoin($other$, $K_{DGK_{eq}}$): 
    $vars_{mutual}$ := copy(getVariables())
    $vars_{mutual}$.retain($other$.getVariables())
    return join($vars_{mutual}$, $this$, $other$, "LEFT", $K_{DGK_{eq}}$)

function minus($other$, $K_{DGK_{eq}}$):
    $vars_{mutual}$ := copy(getVariables())
    $vars_{mutual}$.retain($other$.getVariables())
    $di\textrm{\emph{ff}}$ := {}
    $table_{minus}$ := createBindingsTable()
    $table_{minus}$.setVariables($vars$)
    foreach $var$ : $vars_{mutual}$:
        $bindings_l$ := getBindings($var$)
        $bindings_r$ := $other$.getBindings($var$)
        foreach $(binding, \; patternIDs)$ in $bindings_l$:
            if findPatterns($K_{DGK_{eq}}$, $bindings_r$, $binding$) == :bot:
                $di\textrm{\emph{ff}}$ := $di\textrm{\emph{ff}}$ :U $patternIDs$
    $varBindings_{RND}$ := getBindings(sample($vars_{mutual}$))
    foreach $(binding, \; patternIDs)$ in $varBindings_{RND}$:
        foreach $patternID$ in $patternIDs$:
            if $patternID \in di\textrm{\emph{ff}}$:
                copyBindings($patternID$, $patternID$, $vars_{mutual}$, $this$, $table_{minus}$)
    return $table_{minus}$

function join($vars_{mutual}$, $table_l$, $table_r$, $type$, $K_{DGK_{eq}}$) 
    $vars_l$ := copy($table_l$.getVariables())
    $vars_r$ := copy($table_r$.getVariables())
    $vars$ := $vars_l$ :U $vars_r$
    $vars_l$ := $vars_l$ / $vars_r$
    $vars_r$ := $vars_r$ / $vars_l$
    $table_{join}$ := createBindingsTable()
    $table_{join}$.setVariables($vars$)
    $var$ := sample($vars_{mutual}$) 
    $bindings_l$ = $table_l$.getBindings($var$)
    $bindings_r$ = $table_r$.getBindings($var$)
    foreach $(binding, \; patternIDs_l)$ in $bindings_l$: 
        $patternIDs_r$ := findPatterns($K_{DGK_{eq}}$, $bindings_r$, $binding$)
        if $patternIDs_r$ :neq :bot $\:\:$ or type == "LEFT":
            joinPatterns($vars_{mutual}$, $table_l$, $vars_l$, $patternIDs_l$, $table_r$, $vars_r$, $patternIDs_r$, $table_{join}$, $type$, $K_{DGK_{eq}}$)
    return $table_{join}$

function joinPatterns($vars_{mutual}$, $table_l$, $vars_l$, $patternIDs_l$, $table_r$, $vars_r$, $patternIDs_r$, $table_{join}$, $type$, $K_{DGK_{eq}}$) 
    foreach $patternID_l$ in $patternIDs_l$:
        $match$ := false
        if $patternIDs_r$ :neq :bot:
            foreach $patternID_r$ in $patternIDs_r$:
                if equalPatterns($vars_{mutual}$, $table_l$, $patternID_l$, $table_r$, $patternID_r$, $K_{DGK_{eq}}$): 
                    $match$ := true
                    $patternID$ = generateID()
                    copyBindings($patternID$, $patternID_l$, $vars_{mutual}$, $table_l$, $table_{join}$)
                    copyBindings($patternID$, $patternID_l$, $vars_l$, $table_l$, $table_{join}$)
                    copyBindings($patternID$, $patternID_r$, $vars_r$, $table_r$, $table_{join}$)
        if $match$ == false $\:\:$ and type == "LEFT": 
            copyBindings($patternID_l$, $patternID_l$, $vars_{mutual}$, $table_l$, $table_{join}$)
            copyBindings($patternID_l$, $patternID_l$, $vars_l$, $table_l$, $table_{join}$)

function equalPatterns($vars_{mutual}$, $table_l$, $patternID_l$, $table_r$, $patternID_r$, $K_{DGK_{eq}} = (p, \; vp, \; n, \; u)$) 
    foreach $var$ in $vars_{mutual}$: 
        $binding_l$ = $table_l$.getPatternIDs(var)[$patternID_l$]
        $binding_r$ = $table_r$.getPatternIDs(var)[$patternID_r$]
        if $binding_l$ == :bot $\:\:$ and $binding_r$ :neq :bot
            return false
        elif $binding_l$ :neq :bot $\:\:$ and $binding_r$ == :bot
            return false
        elif $binding_l$ :neq :bot $\:\:$ and $binding_l$ :neq $binding_r$
            return equals($K_{DGK_{eq}}$, $binding_l$, $binding_r$)
    return true

function findPatterns($K_{DGK_{eq}} = (p, \; vp, \; n, \; u)$, $bindings$, $target$):
    $res$ := :bot
    foreach $eqTag$ in $bindings$:
        if equals($K_{DGK_{eq}}$, $eqTag$, $target$) = true:
            $res$ := $eqTag$
    if $eqTagGroup$ :neq :bot:
        return :bot
    return $bindings$[$res$]
