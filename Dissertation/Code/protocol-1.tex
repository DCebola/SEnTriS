function loadState($master$, $rnd$, $det$, $iv$, $schema$):
    $k_{MASTER}$ := $master$ 
    $k_{RND}$ := $rnd$

function exec($T$ = {$t_0$, $t_1$, ... $t_n$}, $schema$):
    if ($schema$ == true)
        encryptSchemaTriples($T$)
    else
        encryptTriples($T$)
    encryptKeywordInfo()

function EncryptSchemaTriples($T$ = {$t_0$, $t_1$, ... $t_n$}):
    foreach $t = (s,p,o)$ in $T$:
        encryptSchemaNode(parseNode($s$))
        encryptSchemaNode(parseNode($p$))
        encryptSchemaNode(parseNode($o$))

function EncryptTriples($T$ = {$t_0$, $t_1$, ... $t_n$}):
	foreach $t = (s,p,o)$ in $T$:
        $freq$ := {}
        $parsed_s$ = parseNode($s$)
        $parsed_p$ = parseNode($p$)
        $parsed_o$ = parseNode($o$)
        $keyword_s$ = parseKeyword($s$)
        $keyword_p$ = parseKeyword($p$)
        $keyword_o$ = parseKeyword($o$)
        $freq$ := $freq$ :U {encryptNode($parsed_p$, "PO" || $keyword_s$)}
        $freq$ := $freq$ :U {encryptNode($parsed_o$, "PO" || $keyword_s$)}
        $freq$ := $freq$ :U {encryptNode($parsed_s$, "SO" || $keyword_p$)}
        $freq$ := $freq$ :U {encryptNode($parsed_o$, "SO" || $keyword_p$)}
        $freq$ := $freq$ :U {encryptNode($parsed_s$, "SP" || $keyword_o$)}
        $freq$ := $freq$ :U {encryptNode($parsed_p$, "SP" || $keyword_o$)}
        $freq$ := $freq$ :U {encryptNode($parsed_s$, "S" || $keyword_p$ || $keyword_o$)}
        $freq$ := $freq$ :U {encryptNode($parsed_p$, "P" || $keyword_s$ || $keyword_o$)}
        $freq$ := $freq$ :U {encryptNode($parsed_o$, "O" || $keyword_s$ || $keyword_p$)}
        encryptTriple("SPO" || $keyword_s$ || $keyword_p$ || $keyword_o$, $freq$)

function encryptKeywordInfo():
    foreach $keyword$ in $keywordFrequencies$:
        $trapdoor$ := ENC($derivedKeys$[$keyword$], $keyword$, $iv_{ZERO}$)
        $encryptedNode$[$trapdoor$] := ENC($k_{RND}$, $keywordFrequencies$[$keyword$])

function encryptSchemaNode($node$):
    $freq$ := incrementKeywordFrequency($schemaKeyword$)
    $trapdoor$ := ENC(deriveKey($schemaKeyword$), $schemaKeyword$, $freq$)
    $encryptedNode$[$trapdoor$] := ENC($k_{RND}$, $node$)
    
function encryptNode($node$, $keyword$):
    $freq$ := incrementKeywordFrequency($keyword$)
    $trapdoor$ := ENC(deriveKey($keyword$), $keyword$, $freq$)
    $encryptedNode$[$trapdoor$] := ENC($k_{RND}$, ENC($k_{DET}$, $node$, $iv_{DET}$))
    return $freq$
 
function encryptTriple($keyword$, $frequencies$): 
    $i$ := 0
    foreach $freq$ in $frequencies$:
        $trapdoor$ := ENC(deriveKey($keyword$), $keyword$, $i$)
        $encryptedNode$[$trapdoor$] := ENC($k_{RND}$, $freq$)
        $i$++

function deriveKey($context$):
    $key$ := $derivedKeys$[$context$]
    if $key$ == :bot :
        $key$ := generateKey($k_{MASTER}$, $context$)
        $derivedKeys$[$context$] := $key$
    return $key$

function incrementKeywordFrequency($keyword$):
    $freq$ := $keywordFrequencies$[$keyword$]
    if $freq$ == :bot :
        $freq$ := 1
        $keywordFrequencies$[$keyword$] := $freq$
    else:
        $keywordFrequencies$[$keyword$] := $freq$ + 1
    return $freq$

function generateTrapdoor($keyword$, $i$):
    return ENC(deriveKey($keyword$), $keyword$, $i$)
    
function generateTrapdoor($keyword$):
    return ENC(deriveKey($keyword$), $keyword$, $iv_{ZERO}$)
	
function generateTrapdoorAndIncrementIV($keyword$):
    return ENC(deriveKey($keyword$), $keyword$, incrementKeywordFrequency($keyword$))
