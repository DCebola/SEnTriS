\makeatletter
\if@todonotes@disabled
    \newcommand{\mytodo}[1]{}
    \newcommand{\myrevise}[3]{#2}
    \newcommand{\mysuggestion}[4]{#2}
    \newcommand{\mycomment}[3]{#2}
\else
    \newcounter{todoCounter}
    \newcommand{\mytodo}[1]
    {
        \stepcounter{todoCounter}
        \sethlcolor{bananayellow}
        \hl{T}\footnotemark[\thetodoCounter]
        \sethlcolor{}
        \renewcommand{\baselinestretch}{1}
        \todo[color={bananayellow}, noline]{\thetodoCounter) \textit{TODO:} #1}
    }
    % \myrevise{author}{text to revise}{explanation}
    \newcounter{notesCounter}
    \newcommand{\myrevise}[3]{
        \stepcounter{notesCounter}
        \sethlcolor{burntsienna}
        \hl{#2}\footnotemark[\thenotesCounter]
        \sethlcolor{}
        \renewcommand{\baselinestretch}{1}
        \todo[color={burntsienna}, noline]{\textbf{\thenotesCounter) [{#1}]:} ~#3}
    }

    % \mysuggestion{author}{text to change}{suggestion}{explanation}
    \newcommand{\mysuggestion}[4]{
        \stepcounter{notesCounter}
        \st{#2}
        \sethlcolor{celadon}
        \hl{#3}\footnotemark[\thenotesCounter]
        \sethlcolor{}
        \renewcommand{\baselinestretch}{1}%
        \todo[color={celadon}, noline]{\textbf{\thenotesCounter) [{#1}]:} ~#4}
    }
    % \mycomment[highlight text]{author}{comment}
    \newcommand{\mycomment}[3]{
        \stepcounter{notesCounter}
        \sethlcolor{babyblue}
        \hl{#2}\footnotemark[\thenotesCounter]
        \sethlcolor{}
        \renewcommand{\baselinestretch}{1}%
        \todo[nolist, color={babyblue}, noline]{\textbf{\thenotesCounter) [{#1}]:} ~#3}
    }
\fi
